388        public SocketState process(long socket) {
389            AjpAprProcessor processor = recycledProcessors.poll();
390            try {
391 
392                if (processor == null) {
393                    processor = createProcessor();
394                }
395 
396                processor.action(ActionCode.ACTION_START, null);
397 
398                if (processor.process(socket)) {
399                    return SocketState.OPEN;
400                } else {
401                    return SocketState.CLOSED;
402                }
403 
404            } catch(java.net.SocketException e) {
405                // SocketExceptions are normal
406                AjpAprProtocol.log.debug
407                    (sm.getString
408                     ("ajpprotocol.proto.socketexception.debug"), e);
409            } catch (java.io.IOException e) {
410                // IOExceptions are normal
411                AjpAprProtocol.log.debug
412                    (sm.getString
413                     ("ajpprotocol.proto.ioexception.debug"), e);
414            }
415            // Future developers: if you discover any other
416            // rare-but-nonfatal exceptions, catch them here, and log as
417            // above.
418            catch (Throwable e) {
419                // any other exception or error is odd. Here we log it
420                // with "ERROR" level, so it will show up even on
421                // less-than-verbose logs.
422                AjpAprProtocol.log.error
423                    (sm.getString("ajpprotocol.proto.error"), e);
424            } finally {
425                processor.action(ActionCode.ACTION_STOP, null);
426                recycledProcessors.offer(processor);
427            }
428            return SocketState.CLOSED;
429        }

443            if (proto.getDomain() != null) {
444                synchronized (this) {
445                    try {
446                        long count = registerCount.incrementAndGet();
447                        RequestInfo rp = processor.getRequest().getRequestProcessor();
448                        rp.setGlobalProcessor(global);
449                        ObjectName rpName = new ObjectName

451                                + proto.getName() + ",name=AjpRequest" + count);
452                        if (log.isDebugEnabled()) {
453                            log.debug("Register " + rpName);
454                        }

456                        rp.setRpName(rpName);

459                    }
