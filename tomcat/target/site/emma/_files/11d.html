532            } catch (IOException e) {
533                // Set error flag
534                error = true;
535            }

551                flush();
552            } catch (IOException e) {
553                // Set error flag
554                error = true;
555            }

562

564                finish();
565            } catch (IOException e) {

567                error = true;
568            }
569 
570        } else if (actionCode == ActionCode.ACTION_START) {
571 
572            started = true;
573 
574        } else if (actionCode == ActionCode.ACTION_STOP) {
575 
576            started = false;
577 
578        } else if (actionCode == ActionCode.ACTION_REQ_SSL_ATTRIBUTE ) {
579 
580            if (!certificates.isNull()) {
581                ByteChunk certData = certificates.getByteChunk();
582                X509Certificate jsseCerts[] = null;
583                ByteArrayInputStream bais =
584                    new ByteArrayInputStream(certData.getBytes(),
585                            certData.getStart(),
586                            certData.getLength());

598                            cf.generateCertificate(bais);

601                            jsseCerts[0] = cert;
602                        } else {

604                            System.arraycopy(jsseCerts,0,temp,0,jsseCerts.length);

606                            jsseCerts = temp;

608                    }
609                } catch (java.security.cert.CertificateException e) {
610                    log.error(sm.getString("ajpprocessor.certs.fail"), e);

612                } catch (NoSuchProviderException e) {
613                    log.error(sm.getString("ajpprocessor.certs.fail"), e);
614                    return;
615                }

617            }

619        } else if (actionCode == ActionCode.ACTION_REQ_HOST_ATTRIBUTE) {

688        byte methodCode = requestHeaderMessage.getByte();
689        if (methodCode != Constants.SC_M_JK_STORED) {
690            String methodName = Constants.methodTransArray[methodCode - 1];
691            request.method().setString(methodName);
692        }
693 
694        requestHeaderMessage.getBytes(request.protocol());
695        requestHeaderMessage.getBytes(request.requestURI());
696 
697        requestHeaderMessage.getBytes(request.remoteAddr());
698        requestHeaderMessage.getBytes(request.remoteHost());
699        requestHeaderMessage.getBytes(request.localName());
700        request.setLocalPort(requestHeaderMessage.getInt());
701 
702        boolean isSSL = requestHeaderMessage.getByte() != 0;
703        if (isSSL) {
704            request.scheme().setString("https");
705        }
706 
707        // Decode headers
708        MimeHeaders headers = request.getMimeHeaders();
709 
710        // Set this every time in case limit has been changed via JMX
711        headers.setLimit(endpoint.getMaxHeaderCount());
712        request.getCookies().setLimit(getMaxCookieCount());
713        
714        boolean contentLengthSet = false;
715        int hCount = requestHeaderMessage.getInt();
716        for(int i = 0 ; i < hCount ; i++) {
717            String hName = null;
718 
719            // Header names are encoded as either an integer code starting
720            // with 0xA0, or as a normal string (in which case the first
721            // two bytes are the length).
722            int isc = requestHeaderMessage.peekInt();
723            int hId = isc & 0xFF;
724 
725            MessageBytes vMB = null;
726            isc &= 0xFF00;
727            if(0xA000 == isc) {
728                requestHeaderMessage.getInt(); // To advance the read position
729                hName = Constants.headerTransArray[hId - 1];
730                vMB = headers.addValue(hName);
731            } else {
732                // reset hId -- if the header currently being read
733                // happens to be 7 or 8 bytes long, the code below
734                // will think it's the content-type header or the
735                // content-length header - SC_REQ_CONTENT_TYPE=7,
736                // SC_REQ_CONTENT_LENGTH=8 - leading to unexpected
737                // behaviour.  see bug 5861 for more information.
738                hId = -1;
739                requestHeaderMessage.getBytes(tmpMB);
740                ByteChunk bc = tmpMB.getByteChunk();
741                vMB = headers.addValue(bc.getBuffer(),
742                        bc.getStart(), bc.getLength());
743            }
744 
745            requestHeaderMessage.getBytes(vMB);
746 
747            if (hId == Constants.SC_REQ_CONTENT_LENGTH ||
748                    (hId == -1 && tmpMB.equalsIgnoreCase("Content-Length"))) {
749                // just read the content-length header, so set it
750                long cl = vMB.getLong();
751                if (contentLengthSet) {
752                    response.setStatus(HttpServletResponse.SC_BAD_REQUEST);
753                    error = true;
754                } else {
755                    contentLengthSet = true;
756                    // Set the content-length header for the request
757                    if(cl < Integer.MAX_VALUE)
758                        request.setContentLength( (int)cl );
759                }
760            } else if (hId == Constants.SC_REQ_CONTENT_TYPE ||
761                    (hId == -1 && tmpMB.equalsIgnoreCase("Content-Type"))) {
762                // just read the content-type header, so set it
763                ByteChunk bchunk = vMB.getByteChunk();

765                        bchunk.getOffset(),
766                        bchunk.getLength());
767            }
768        }

771        boolean secret = false;
772        byte attributeCode;

774                != Constants.SC_A_ARE_DONE) {
775 
776            switch (attributeCode) {
777

780                String n = tmpMB.toString();
781                requestHeaderMessage.getBytes(tmpMB);
782                String v = tmpMB.toString();
783                /*
784                 * AJP13 misses to forward the local IP address and the
785                 * remote port. Allow the AJP connector to add this info via
786                 * private request attributes.
787                 * We will accept the forwarded data and remove it from the
788                 * public list of request attributes.
789                 */
790                if(n.equals(Constants.SC_A_REQ_LOCAL_ADDR)) {
791                    request.localAddr().setString(v);
792                } else if(n.equals(Constants.SC_A_REQ_REMOTE_PORT)) {
793                    try {
794                        request.setRemotePort(Integer.parseInt(v));
795                    } catch (NumberFormatException nfe) {
796                    }
797                } else {
798                    request.setAttribute(n, v );
799                }
800                break;
801 
802            case Constants.SC_A_CONTEXT :
803                requestHeaderMessage.getBytes(tmpMB);
804                // nothing
805                break;
806 
807            case Constants.SC_A_SERVLET_PATH :
808                requestHeaderMessage.getBytes(tmpMB);
809                // nothing
810                break;
811 
812            case Constants.SC_A_REMOTE_USER :
813                if (tomcatAuthentication) {
814                    // ignore server
815                    requestHeaderMessage.getBytes(tmpMB);
816                } else {
817                    requestHeaderMessage.getBytes(request.getRemoteUser());
818                }
819                break;
820 
821            case Constants.SC_A_AUTH_TYPE :
822                if (tomcatAuthentication) {
823                    // ignore server
824                    requestHeaderMessage.getBytes(tmpMB);
825                } else {
826                    requestHeaderMessage.getBytes(request.getAuthType());
827                }
828                break;
829 
830            case Constants.SC_A_QUERY_STRING :
831                requestHeaderMessage.getBytes(request.queryString());
832                break;
833 
834            case Constants.SC_A_JVM_ROUTE :
835                requestHeaderMessage.getBytes(request.instanceId());
836                break;
837 
838            case Constants.SC_A_SSL_CERT :
839                request.scheme().setString("https");
840                // SSL certificate extraction is lazy, moved to JkCoyoteHandler
841                requestHeaderMessage.getBytes(certificates);
842                break;
843 
844            case Constants.SC_A_SSL_CIPHER :
845                request.scheme().setString("https");
846                requestHeaderMessage.getBytes(tmpMB);
847                request.setAttribute(JIoEndpoint.CIPHER_SUITE_KEY,
848                                     tmpMB.toString());
849                break;
850 
851            case Constants.SC_A_SSL_SESSION :
852                request.scheme().setString("https");
853                requestHeaderMessage.getBytes(tmpMB);
854                request.setAttribute(JIoEndpoint.SESSION_ID_KEY,
855                                     tmpMB.toString());
856                break;
857 
858            case Constants.SC_A_SSL_KEY_SIZE :
859                request.setAttribute(JIoEndpoint.KEY_SIZE_KEY,
860                                     new Integer(requestHeaderMessage.getInt()));
861                break;
862 
863            case Constants.SC_A_STORED_METHOD:
864                requestHeaderMessage.getBytes(request.method());
865                break;
866 
867            case Constants.SC_A_SECRET:
868                requestHeaderMessage.getBytes(tmpMB);
869                if (requiredSecret != null) {
870                    secret = true;
871                    if (!tmpMB.equals(requiredSecret)) {
872                        response.setStatus(403);
873                        error = true;
874                    }
875                }
876                break;
877 
878            default:
879                // Ignore unknown attribute for backward compatibility
880                break;
881 
882            }
883 
884        }
885 
886        // Check if secret was submitted if required
887        if ((requiredSecret != null) && !secret) {
888            response.setStatus(403);
889            error = true;
890        }
891 
892        // Check for a full URI (including protocol://host:port/)
893        ByteChunk uriBC = request.requestURI().getByteChunk();
894        if (uriBC.startsWithIgnoreCase("http", 0)) {
895 
896            int pos = uriBC.indexOf("://", 0, 3, 4);
897            int uriBCStart = uriBC.getStart();
898            int slashPos = -1;
899            if (pos != -1) {
900                byte[] uriB = uriBC.getBytes();
901                slashPos = uriBC.indexOf('/', pos + 3);
902                if (slashPos == -1) {
903                    slashPos = uriBC.getLength();
904                    // Set URI as "/"
905                    request.requestURI().setBytes
906                        (uriB, uriBCStart + pos + 1, 1);
907                } else {
908                    request.requestURI().setBytes
909                        (uriB, uriBCStart + slashPos,
910                         uriBC.getLength() - slashPos);
911                }
912                MessageBytes hostMB = headers.setValue("host");
913                hostMB.setBytes(uriB, uriBCStart + pos + 3,
914                                slashPos - pos - 3);
915            }
916 
917        }
918 
919        MessageBytes valueMB = request.getMimeHeaders().getValue("host");
920        parseHost(valueMB);
921 
922        if (error) {
923            adapter.log(request, response, 0);
924        }
925    }

928    /**
929     * Parse host.
930     */
931    public void parseHost(MessageBytes valueMB) {
932 
933        if (valueMB == null || valueMB.isNull()) {
934            // HTTP/1.0
935            request.setServerPort(request.getLocalPort());
936            try {
937                request.serverName().duplicate(request.localName());
938            } catch (IOException e) {
939                response.setStatus(400);
940                error = true;
941            }
942            return;
943        }
944 
945        ByteChunk valueBC = valueMB.getByteChunk();
946        byte[] valueB = valueBC.getBytes();
947        int valueL = valueBC.getLength();
948        int valueS = valueBC.getStart();
949        int colonPos = -1;
950        if (hostNameC.length < valueL) {
951            hostNameC = new char[valueL];
952        }
953 
954        boolean ipv6 = (valueB[valueS] == '[');
955        boolean bracketClosed = false;
956        for (int i = 0; i < valueL; i++) {
957            char b = (char) valueB[i + valueS];
958            hostNameC[i] = b;
959            if (b == ']') {
960                bracketClosed = true;
961            } else if (b == ':') {
962                if (!ipv6 || bracketClosed) {
963                    colonPos = i;
964                    break;
965                }
966            }
967        }
968 
969        if (colonPos < 0) {
970            if (request.scheme().equalsIgnoreCase("https")) {
971                // 443 - Default HTTPS port
972                request.setServerPort(443);
973            } else {
974                // 80 - Default HTTTP port
975                request.setServerPort(80);
976            }
977            request.serverName().setChars(hostNameC, 0, valueL);
978        } else {
979 
980            request.serverName().setChars(hostNameC, 0, colonPos);
981 
982            int port = 0;
983            int mult = 1;
984            for (int i = valueL - 1; i > colonPos; i--) {
985                int charValue = HexUtils.getDec(valueB[i + valueS]);
986                if (charValue == -1) {
987                    // Invalid character
988                    error = true;
989                    // 400 - Bad request
990                    response.setStatus(400);
991                    break;
992                }
993                port = port + (charValue * mult);
994                mult = 10 * mult;
995            }
996            request.setServerPort(port);
997 
998        }

1003    /**
1004     * When committing the response, we have to validate the set of headers, as

1014

1017        String message = null;
1018        if (org.apache.coyote.Constants.USE_CUSTOM_STATUS_MSG_IN_HEADER &&
1019                HttpMessages.isSafeInHttpHeader(response.getMessage())) {
1020            message = response.getMessage();
1021        }
1022        if (message == null){
1023            message = HttpMessages.getMessage(response.getStatus());
1024        }
1025        if (message == null) {
1026            // mod_jk + httpd 2.x fails with a null status message - bug 45026
1027            message = Integer.toString(response.getStatus());
1028        }
1029        tmpMB.setString(message);
1030        responseHeaderMessage.appendBytes(tmpMB);
1031 
1032        // Special headers
1033        MimeHeaders headers = response.getMimeHeaders();
1034        String contentType = response.getContentType();
1035        if (contentType != null) {
1036            headers.setValue("Content-Type").setString(contentType);
1037        }
1038        String contentLanguage = response.getContentLanguage();
1039        if (contentLanguage != null) {
1040            headers.setValue("Content-Language").setString(contentLanguage);
1041        }
1042        long contentLength = response.getContentLengthLong();
1043        if (contentLength >= 0) {
1044            headers.setValue("Content-Length").setLong(contentLength);
1045        }
1046 
1047        // Other headers
1048        int numHeaders = headers.size();

1050        for (int i = 0; i < numHeaders; i++) {
1051            MessageBytes hN = headers.getName(i);
1052            int hC = Constants.getResponseAjpIndex(hN.toString());
1053            if (hC > 0) {
1054                responseHeaderMessage.appendInt(hC);
1055            }
1056            else {
1057                responseHeaderMessage.appendBytes(hN);
1058            }
1059            MessageBytes hV=headers.getValue(i);
1060            responseHeaderMessage.appendBytes(hV);
1061        }
1062

1064        responseHeaderMessage.end();
1065        output.write(responseHeaderMessage.getBuffer(), 0, responseHeaderMessage.getLen());
1066
