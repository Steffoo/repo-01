402                // any other exception or error is odd. Here we log it
403                // with "ERROR" level, so it will show up even on
404                // less-than-verbose logs.
405                AjpProtocol.log.error
406                    (sm.getString("ajpprotocol.proto.error"), e);
407            } finally {
408                processor.action(ActionCode.ACTION_STOP, null);
409                recycledProcessors.offer(processor);
410            }
411            return false;
412        }

426        protected void register(AjpProcessor processor) {
427            if (proto.getDomain() != null) {
428                synchronized (this) {
429                    try {
430                        long count = registerCount.incrementAndGet();
431                        RequestInfo rp = processor.getRequest().getRequestProcessor();
432                        rp.setGlobalProcessor(global);
433                        ObjectName rpName = new ObjectName
434                            (proto.getDomain() + ":type=RequestProcessor,worker="
435                                + proto.getName() + ",name=AjpRequest" + count);
436                        if (log.isDebugEnabled()) {
437                            log.debug("Register " + rpName);
438                        }
439                        Registry.getRegistry(null, null).registerComponent(rp, rpName, null);
440                        rp.setRpName(rpName);
441                    } catch (Exception e) {
442                        log.warn("Error registering request");
443                    }
444                }
445            }
