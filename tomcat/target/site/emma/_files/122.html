600    /**
601     * Reallocates buffer since the spec requires it to be unbounded.
602     */
603    private void reAllocBuff(int len) {
604        
605        if (bufferSize + len <= cb.length) {
606            bufferSize = cb.length;
607            return;
608        }
609        
610        if (len < cb.length) {
611            len = cb.length;
612        }
613        
614        char[] tmp = new char[cb.length + len];
615        System.arraycopy(cb, 0, tmp, 0, cb.length);
616        cb = tmp;
617        bufferSize = cb.length;
618    }
