576    public int receive( Msg msg, MsgContext ep )

580        }

582        byte buf[]=msg.getBuffer();
583        int hlen=msg.getHeaderLength();
584        
585        // XXX If the length in the packet header doesn't agree with the
586        // actual number of bytes read, it should probably return an error
587        // value.  Also, callers of this method never use the length
588        // returned -- should probably return true/false instead.

595            return rd;

598        msg.processHeader();
599 
600        /* After processing the header we know the body
601           length
602        */
603        int blen=msg.getLen();
604        
605        // XXX check if enough space - it's assert()-ed !!!
606        
607        int total_read = 0;
608        
609        total_read = this.read(ep, buf, hlen, blen);
610        
611        if ((total_read <= 0) && (blen > 0)) {
612            log.warn("can't read body, waited #" + blen);
613            return  -1;
614        }
615        
616        if (total_read != blen) {
617             log.warn( "incomplete read, waited #" + blen +
618                        " got only " + total_read);

620        }
621

625    /**
626     * Read N bytes from the InputStream, and ensure we got them all
627     * Under heavy load we could experience many fragmented packets
628     * just read Unix Network Programming to recall that a call to
629     * read didn't ensure you got all the data you want
630     *
631     * from read() Linux manual
632     *
633     * On success, the number of bytes read is returned (zero indicates end
634     * of file),and the file position is advanced by this number.
635     * It is not an error if this number is smaller than the number of bytes
636     * requested; this may happen for example because fewer bytes
637     * are actually available right now (maybe because we were close to
638     * end-of-file, or because we are reading from a pipe, or  from  a
639     * terminal),  or  because  read()  was interrupted by a signal.
640     * On error, -1 is returned, and errno is set appropriately. In this
641     * case it is left unspecified whether the file position (if any) changes.
642     *
643     **/
644    public int read( MsgContext ep, byte[] b, int offset, int len)
645        throws IOException
646    {
647        InputStream is=(InputStream)ep.getNote( isNote );
648        int pos = 0;
649        int got;
650 
651        while(pos < len) {
652            try {
653                got = is.read(b, pos + offset, len - pos);
654            } catch(ClosedChannelException sex) {
655                if(pos > 0) {
656                    log.info("Error reading data after "+pos+"bytes",sex);
657                } else {
658                    log.debug("Error reading data", sex);
659                }
660                got = -1;
661            }
662            if (log.isTraceEnabled()) {
663                log.trace("read() " + b + " " + (b==null ? 0: b.length) + " " +
664                          offset + " " + len + " = " + got );
665            }
666 
667            // connection just closed by remote. 
668            if (got <= 0) {
669                // This happens periodically, as apache restarts
670                // periodically.
671                // It should be more gracefull ! - another feature for Ajp14
672                // log.warn( "server has closed the current connection (-1)" );
673                return -3;
674            }
675 
676            pos += got;
677        }
678        return pos;
679    }

772        byte serverAddr[] = server.getAddress();

778            if (serverAddr[i] != clientAddr[i]) {
779                match = false;
780                break;

782        }

784            return (true);

787        for (int i = 0; i < serverAddr.length; i++) {
788            if (serverAddr[i] != clientAddr[(serverAddr.length-1)-i])
789                return (false);
