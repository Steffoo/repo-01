480                 * remote port. Allow the AJP connector to add this info via
481                 * private request attributes.
482                 * We will accept the forwarded data and remove it from the
483                 * public list of request attributes.
484                 */
485                if(n.equals(AjpConstants.SC_A_REQ_LOCAL_ADDR)) {
486                    req.localAddr().setString(v);
487                } else if(n.equals(AjpConstants.SC_A_REQ_REMOTE_PORT)) {
488                    try {
489                        req.setRemotePort(Integer.parseInt(v));
490                    } catch (NumberFormatException nfe) {
491                    }
492                } else {
493                    req.setAttribute(n, v );
494                    if(log.isTraceEnabled())
495                        log.trace("jk Attribute set " + n + "=" + v);
496                }
497            }
498 
499 
500            // 1 string attributes
501            switch(attributeCode) {
502            case AjpConstants.SC_A_CONTEXT      :
503                msg.getBytes( tmpMB );
504                // nothing
505                break;
506                
507            case AjpConstants.SC_A_SERVLET_PATH :
508                msg.getBytes( tmpMB );
509                // nothing 
510                break;
511                
512            case AjpConstants.SC_A_REMOTE_USER  :
513                if( tomcatAuthentication ) {
514                    // ignore server
515                    msg.getBytes( tmpMB );
516                } else {
517                    msg.getBytes(req.getRemoteUser());
518                }
519                break;
520                
521            case AjpConstants.SC_A_AUTH_TYPE    :
522                if( tomcatAuthentication ) {
523                    // ignore server
524                    msg.getBytes( tmpMB );
525                } else {
526                    msg.getBytes(req.getAuthType());
527                }
528                break;
529                
530            case AjpConstants.SC_A_QUERY_STRING :
531                msg.getBytes(req.queryString());
532                break;
533                
534            case AjpConstants.SC_A_JVM_ROUTE    :
535                msg.getBytes(req.instanceId());
536                break;
537                
538            case AjpConstants.SC_A_SSL_CERT     :
539                req.scheme().setString( "https" );
540                // Transform the string into certificate.
541                MessageBytes tmpMB2 = (MessageBytes)req.getNote(WorkerEnv.SSL_CERT_NOTE);
542                if(tmpMB2 == null) {
543                    tmpMB2 = MessageBytes.newInstance();
544                    req.setNote(WorkerEnv.SSL_CERT_NOTE, tmpMB2);
545                }
546                // SSL certificate extraction is costy, moved to JkCoyoteHandler
547                msg.getBytes(tmpMB2);
548                break;
549                
550            case AjpConstants.SC_A_SSL_CIPHER   :
551                req.scheme().setString( "https" );
552                msg.getBytes(tmpMB);
553                req.setAttribute(SSLSupport.CIPHER_SUITE_KEY,
554                                 tmpMB.toString());
555                break;
556                
557            case AjpConstants.SC_A_SSL_SESSION  :
558                req.scheme().setString( "https" );
559                msg.getBytes(tmpMB);
560                req.setAttribute(SSLSupport.SESSION_ID_KEY, 
561                                  tmpMB.toString());
562                break;
563                
564            case AjpConstants.SC_A_SECRET  :
565                msg.getBytes(tmpMB);
566                String secret=tmpMB.toString();
567                if(log.isTraceEnabled())
568                    log.trace("Secret: " + secret );
569                // endpoint note
570                ep.setNote( secretNote, secret );
571                break;
572                
573            case AjpConstants.SC_A_STORED_METHOD:
574                msg.getBytes(req.method()); 
575                break;
576                
577            default:
578                break; // ignore, we don't know about it - backward compat
579            }
580        }
581        return 200;
582    }

584    private void decodeHeaders( MsgContext ep, Msg msg, Request req,
585                                MessageBytes tmpMB ) {
586        // Decode headers
587        MimeHeaders headers = req.getMimeHeaders();
588 
589        // Set this every time in case limit has been changed via JMX
590        req.getMimeHeaders().setLimit(maxHeaderCount);
591 
592        int hCount = msg.getInt();
593        for(int i = 0 ; i < hCount ; i++) {
594            String hName = null;
595 
596            // Header names are encoded as either an integer code starting
597            // with 0xA0, or as a normal string (in which case the first
598            // two bytes are the length).
599            int isc = msg.peekInt();
600            int hId = isc & 0xFF;
601 
602            MessageBytes vMB=null;
603            isc &= 0xFF00;
604            if(0xA000 == isc) {
605                msg.getInt(); // To advance the read position
606                hName = AjpConstants.headerTransArray[hId - 1];
607                vMB=headers.addValue( hName );
608            } else {
609                // reset hId -- if the header currently being read
610                // happens to be 7 or 8 bytes long, the code below
611                // will think it's the content-type header or the
612                // content-length header - SC_REQ_CONTENT_TYPE=7,
613                // SC_REQ_CONTENT_LENGTH=8 - leading to unexpected
614                // behaviour.  see bug 5861 for more information.
615                hId = -1;
616                msg.getBytes( tmpMB );
617                ByteChunk bc=tmpMB.getByteChunk();
618                vMB=headers.addValue( bc.getBuffer(),
619                                      bc.getStart(), bc.getLength() );
620            }
621 
622            msg.getBytes(vMB);
623 
624            if (hId == AjpConstants.SC_REQ_CONTENT_LENGTH ||
625                (hId == -1 && tmpMB.equalsIgnoreCase("Content-Length"))) {
626                // just read the content-length header, so set it
627                long cl = vMB.getLong();
628                if(cl < Integer.MAX_VALUE) 
629                    req.setContentLength( (int)cl );
630            } else if (hId == AjpConstants.SC_REQ_CONTENT_TYPE ||
631                (hId == -1 && tmpMB.equalsIgnoreCase("Content-Type"))) {
632                // just read the content-type header, so set it
633                ByteChunk bchunk = vMB.getByteChunk();
634                req.contentType().setBytes(bchunk.getBytes(),
635                                           bchunk.getOffset(),
636                                           bchunk.getLength());
637            }
638        }
639    }

641    /**
642     * Parse host.
643     */
644    private void parseHost(MessageBytes valueMB, Request request) 
645        throws IOException {
646 
647        if (valueMB == null || valueMB.isNull()) {
648            // HTTP/1.0
649            // Default is what the socket tells us. Overriden if a host is 
650            // found/parsed
651            request.setServerPort(request.getLocalPort());
652            request.serverName().duplicate(request.localName());
653            return;
654        }
655 
656        ByteChunk valueBC = valueMB.getByteChunk();
657        byte[] valueB = valueBC.getBytes();
658        int valueL = valueBC.getLength();
659        int valueS = valueBC.getStart();
660        int colonPos = -1;
661        CharChunk hostNameC = (CharChunk)request.getNote(HOSTBUFFER);
662        if(hostNameC == null) {
663            hostNameC = new CharChunk(valueL);
664            request.setNote(HOSTBUFFER, hostNameC);
665        }
666        hostNameC.recycle();
667 
668        boolean ipv6 = (valueB[valueS] == '[');
669        boolean bracketClosed = false;
670        for (int i = 0; i < valueL; i++) {
671            char b = (char) valueB[i + valueS];
672            hostNameC.append(b);
673            if (b == ']') {
674                bracketClosed = true;
675            } else if (b == ':') {
676                if (!ipv6 || bracketClosed) {
677                    colonPos = i;
678                    break;
679                }
680            }
681        }
682 
683        if (colonPos < 0) {
684            if (request.scheme().equalsIgnoreCase("https")) {
685                // 80 - Default HTTTP port
686                request.setServerPort(443);
687            } else {
688                // 443 - Default HTTPS port
689                request.setServerPort(80);
690            }
691            request.serverName().setChars(hostNameC.getChars(), 
692                                          hostNameC.getStart(), 
693                                          hostNameC.getLength());
694        } else {
695 
696            request.serverName().setChars(hostNameC.getChars(), 
697                                          hostNameC.getStart(), colonPos);
698 
699            int port = 0;
700            int mult = 1;
701            for (int i = valueL - 1; i > colonPos; i--) {
702                int charValue = HexUtils.getDec(valueB[i + valueS]);
703                if (charValue == -1) {
704                    // Invalid character
705                    throw new CharConversionException("Invalid char in port: " + valueB[i + valueS]); 
706                }
707                port = port + (charValue * mult);
708                mult = 10 * mult;
709            }
710            request.setServerPort(port);
711 
712        }
713 
714    }
