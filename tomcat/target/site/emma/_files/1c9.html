735    private void writeObject(ObjectOutput stream) throws IOException {
736        // Write the scalar instance variables (except Manager)
737        stream.writeObject(new Long(creationTime));
738        stream.writeObject(new Long(lastAccessedTime));
739        stream.writeObject(new Integer(maxInactiveInterval));
740        stream.writeObject(new Boolean(isNew));
741        stream.writeObject(new Boolean(isValid));
742        stream.writeObject(new Long(thisAccessedTime));
743        stream.writeObject(new Long(version));
744        stream.writeBoolean(getPrincipal() != null);
745        if (getPrincipal() != null) {
746            SerializablePrincipal.writePrincipal((GenericPrincipal) principal,stream);
747        }
748 
749        stream.writeObject(id);
750        if (log.isDebugEnabled()) log.debug(sm.getString("deltaSession.writeSession", id));
751 
752        // Accumulate the names of serializable and non-serializable attributes
753        String keys[] = keys();
754        ArrayList saveNames = new ArrayList();
755        ArrayList saveValues = new ArrayList();
756        for (int i = 0; i < keys.length; i++) {
757            Object value = null;
758            value = attributes.get(keys[i]);
759            if (value != null && !exclude(keys[i], value) &&
760                    isAttributeDistributable(keys[i], value)) {
761                saveNames.add(keys[i]);
762                saveValues.add(value);
763            }
764        }
765 
766        // Serialize the attribute count and the Serializable attributes
767        int n = saveNames.size();
768        stream.writeObject(new Integer(n));
769        for (int i = 0; i < n; i++) {
770            stream.writeObject( (String) saveNames.get(i));
771            try {
772                stream.writeObject(saveValues.get(i));
773            } catch (NotSerializableException e) {
774                log.error(sm.getString("standardSession.notSerializable",saveNames.get(i), id), e);
775                stream.writeObject(NOT_SERIALIZED);
776                log.error("  storing attribute '" + saveNames.get(i)+ "' with value NOT_SERIALIZED");
777            }
778        }
