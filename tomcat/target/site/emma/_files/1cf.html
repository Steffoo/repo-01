423                rs.expire(false);  //cleans up the previous values, since we are not doing removes
424                session = null;
425            }//end if
426 
427            if (session==null) {
428                session = createSession(null,false, false);
429                sessions.remove(session.getIdInternal());
430            }
431            
432            
433            boolean hasPrincipal = session_in.readBoolean();
434            SerializablePrincipal p = null;
435            if ( hasPrincipal )
436                p = (SerializablePrincipal)session_in.readObject();
437            ((ReplicatedSession)session).readObjectData(session_in);
438            if ( hasPrincipal )
439                session.setPrincipal(p.getPrincipal(getContainer().getRealm()));
440            ((ReplicatedSession)session).setId(sessionId,isNew);
441            ReplicatedSession rsession = (ReplicatedSession)session; 
442            rsession.setAccessCount(1);
443            session.setManager(this);
444            session.setValid(true);
445            rsession.setLastAccessedTime(System.currentTimeMillis());
446            rsession.setThisAccessedTime(System.currentTimeMillis());
447            ((ReplicatedSession)session).setAccessCount(0);
448            session.setNew(false);
449            if(log.isTraceEnabled())
450                 log.trace("Session loaded id="+sessionId +
451                               " actualId="+session.getId()+ 
452                               " exists="+this.sessions.containsKey(sessionId)+
453                               " valid="+rsession.isValid());
454            return session;
455 
456        }
457        catch ( Exception x )
458        {
459            log.error("Failed to deserialize the session!",x);
460        }
461        return null;
462    }

468     * Prepare for the beginning of active use of the public methods of this

471     * Starts the cluster communication channel, this will connect with the other nodes
472     * in the cluster, and request the current session state to be transferred to this node.
473     * @exception IllegalStateException if this component has already been
474     *  started
475     * @exception LifecycleException if this component detects a fatal error
476     *  that prevents this component from being used

478    public void start() throws LifecycleException {
479        mManagerRunning = true;
480        super.start();
481        try {
482            //the channel is already running
483            if ( mChannelStarted ) return;
484            if(log.isInfoEnabled())
485                log.info("Starting clustering manager...:"+getName());
486            if ( cluster == null ) {
487                log.error("Starting... no cluster associated with this context:"+getName());
488                return;
489            }
490            cluster.registerManager(this);
491 
492            if (cluster.getMembers().length > 0) {
493                Member mbr = cluster.getMembers()[0];
494                SessionMessage msg =
495                    new SessionMessageImpl(this.getName(),
496                                       SessionMessage.EVT_GET_ALL_SESSIONS,
497                                       null,
498                                       "GET-ALL",
499                                       "GET-ALL-"+this.getName());
500                cluster.send(msg, mbr);
501                if(log.isWarnEnabled())
502                     log.warn("Manager["+getName()+"], requesting session state from "+mbr+
503                         ". This operation will timeout if no session state has been received within "+
504                         "60 seconds");
505                long reqStart = System.currentTimeMillis();
506                long reqNow = 0;
507                boolean isTimeout=false;
508                do {
509                    try {
510                        Thread.sleep(100);
511                    }catch ( Exception sleep) {}
512                    reqNow = System.currentTimeMillis();
513                    isTimeout=((reqNow-reqStart)>(1000*60));
514                } while ( (!isStateTransferred()) && (!isTimeout));
515                if ( isTimeout || (!isStateTransferred()) ) {
516                    log.error("Manager["+getName()+"], No session state received, timing out.");
517                }else {
518                    if(log.isInfoEnabled())
519                        log.info("Manager["+getName()+"], session state received in "+(reqNow-reqStart)+" ms.");
520                }
521            } else {
522                if(log.isInfoEnabled())
523                    log.info("Manager["+getName()+"], skipping state transfer. No members active in cluster group.");
524            }//end if
525            mChannelStarted = true;
526        }  catch ( Exception x ) {
527            log.error("Unable to start SimpleTcpReplicationManager",x);
528        }
529    }

564    /**
565     * This method is called by the received thread when a SessionMessage has
566     * been received from one of the other nodes in the cluster.
567     * @param msg - the message received
568     * @param sender - the sender of the message, this is used if we receive a
569     *                 EVT_GET_ALL_SESSION message, so that we only reply to
570     *                 the requesting node
571     */
572    protected void messageReceived( SessionMessage msg, Member sender ) {
573        try  {
574            if(log.isInfoEnabled()) {
575                log.debug("Received SessionMessage of type="+msg.getEventTypeString());
576                log.debug("Received SessionMessage sender="+sender);
577            }
578            switch ( msg.getEventType() ) {
579                case SessionMessage.EVT_GET_ALL_SESSIONS: {
580                    //get a list of all the session from this manager
581                    Object[] sessions = findSessions();
582                    java.io.ByteArrayOutputStream bout = new java.io.ByteArrayOutputStream();
583                    java.io.ObjectOutputStream oout = new java.io.ObjectOutputStream(bout);
584                    oout.writeInt(sessions.length);
585                    for (int i=0; i<sessions.length; i++){
586                        ReplicatedSession ses = (ReplicatedSession)sessions[i];
587                        oout.writeUTF(ses.getIdInternal());
588                        byte[] data = writeSession(ses);
589                        oout.writeObject(data);
590                    }//for
591                    //don't send a message if we don't have to
592                    oout.flush();
593                    oout.close();
594                    byte[] data = bout.toByteArray();
595                    SessionMessage newmsg = new SessionMessageImpl(name,
596                        SessionMessage.EVT_ALL_SESSION_DATA,
597                        data, "SESSION-STATE","SESSION-STATE-"+getName());
598                    cluster.send(newmsg, sender);
599                    break;
600                }
601                case SessionMessage.EVT_ALL_SESSION_DATA: {
602                    java.io.ByteArrayInputStream bin =
603                        new java.io.ByteArrayInputStream(msg.getSession());
604                    java.io.ObjectInputStream oin = new java.io.ObjectInputStream(bin);
605                    int size = oin.readInt();
606                    for ( int i=0; i<size; i++) {
607                        String id = oin.readUTF();
608                        byte[] data = (byte[])oin.readObject();
609                        Session session = readSession(data,id);
610                    }//for
611                    stateTransferred=true;
612                    break;
613                }
614                case SessionMessage.EVT_SESSION_CREATED: {
615                    Session session = this.readSession(msg.getSession(),msg.getSessionID());
616                    if ( log.isDebugEnabled() ) {
617                        log.debug("Received replicated session=" + session +
618                            " isValid=" + session.isValid());
619                    }
620                    break;
621                }
622                case SessionMessage.EVT_SESSION_EXPIRED: {
623                    Session session = findSession(msg.getSessionID());
624                    if ( session != null ) {
625                        session.expire();
626                        this.remove(session);
627                    }//end if
628                    break;
629                }
630                case SessionMessage.EVT_SESSION_ACCESSED :{
631                    Session session = findSession(msg.getSessionID());
632                    if ( session != null ) {
633                        session.access();
634                        session.endAccess();
635                    }
636                    break;
637                }
638                default:  {
639                    //we didn't recognize the message type, do nothing
640                    break;
641                }
642            }//switch
643        }
644        catch ( Exception x )
645        {
646            log.error("Unable to receive message through TCP channel",x);
647        }
648    }
