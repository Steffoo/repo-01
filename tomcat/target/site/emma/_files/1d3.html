274        args = arg;
275        String tok;
276 
277        dieLevel = NO_DIE_LEVEL;
278        die = dieLevel;
279 
280        while ((tok = nextArg()) != null) {
281            if (tok.equals(SWITCH_VERBOSE)) {
282                verbose = true;
283                showSuccess = true;
284                listErrors = true;
285            } else if (tok.equals(SWITCH_OUTPUT_DIR)) {
286                tok = nextArg();
287                setOutputDir( tok );
288            } else if (tok.equals(SWITCH_PACKAGE_NAME)) {
289                targetPackage = nextArg();
290            } else if (tok.equals(SWITCH_COMPILE)) {
291                compile=true;
292            } else if (tok.equals(SWITCH_CLASS_NAME)) {
293                targetClassName = nextArg();
294            } else if (tok.equals(SWITCH_URI_BASE)) {
295                uriBase=nextArg();
296            } else if (tok.equals(SWITCH_URI_ROOT)) {
297                setUriroot( nextArg());
298            } else if (tok.equals(SWITCH_FILE_WEBAPP)) {
299                setUriroot( nextArg());
300            } else if ( tok.equals( SHOW_SUCCESS ) ) {
301                showSuccess = true;
302            } else if ( tok.equals( LIST_ERRORS ) ) {
303                listErrors = true;
304            } else if (tok.equals(SWITCH_WEBAPP_INC)) {
305                webxmlFile = nextArg();
306                if (webxmlFile != null) {
307                    webxmlLevel = INC_WEBXML;
308                }
309            } else if (tok.equals(SWITCH_WEBAPP_XML)) {
310                webxmlFile = nextArg();
311                if (webxmlFile != null) {
312                    webxmlLevel = ALL_WEBXML;
313                }
314            } else if (tok.equals(SWITCH_WEBAPP_XML_ENCODING)) {
315                setWebXmlEncoding(nextArg());
316            } else if (tok.equals(SWITCH_ADD_WEBAPP_XML_MAPPINGS)) {
317                setAddWebXmlMappings(true);
318            } else if (tok.equals(SWITCH_MAPPED)) {
319                mappedFile = true;
320            } else if (tok.equals(SWITCH_XPOWERED_BY)) {
321                xpoweredBy = true;
322            } else if (tok.equals(SWITCH_TRIM_SPACES)) {
323                setTrimSpaces(true);
324            } else if (tok.equals(SWITCH_CACHE)) {
325                tok = nextArg();
326                if ("false".equals(tok)) {
327                    caching = false;
328                } else {
329                    caching = true;
330                }            
331            } else if (tok.equals(SWITCH_CLASSPATH)) {
332                setClassPath(nextArg());
333            } else if (tok.startsWith(SWITCH_DIE)) {
334                try {
335                    dieLevel = Integer.parseInt(
336                        tok.substring(SWITCH_DIE.length()));
337                } catch (NumberFormatException nfe) {
338                    dieLevel = DEFAULT_DIE_LEVEL;
339                }
340                die = dieLevel;
341            } else if (tok.equals(SWITCH_HELP)) {
342                helpNeeded = true;
343            } else if (tok.equals(SWITCH_POOLING)) {
344                tok = nextArg();
345                if ("false".equals(tok)) {
346                    poolingEnabled = false;
347                } else {
348                    poolingEnabled = true;
349                }
350            } else if (tok.equals(SWITCH_ENCODING)) {
351                setJavaEncoding(nextArg());
352            } else if (tok.equals(SWITCH_SOURCE)) {
353                setCompilerSourceVM(nextArg());
354            } else if (tok.equals(SWITCH_TARGET)) {
355                setCompilerTargetVM(nextArg());
356            } else if (tok.equals(SWITCH_SMAP)) {
357                smapSuppressed = false;
358            } else if (tok.equals(SWITCH_DUMP_SMAP)) {
359                smapDumped = true;
360            } else if (tok.equals(SWITCH_VALIDATE_TLD)) {
361                setValidateTld(true);
362            } else if (tok.equals(SWITCH_VALIDATE_XML)) {
363                setValidateXml(true);
364            } else if (tok.equals(SWITCH_BLOCK_EXTERNAL)) {
365                setBlockExternal(true);
366            } else if (tok.equals(SWITCH_NO_BLOCK_EXTERNAL)) {
367                setBlockExternal(false);
368            } else {
369                if (tok.startsWith("-")) {
370                    throw new JasperException("Unrecognized option: " + tok +
371                        ".  Use -help for help.");
372                }
373                if (!fullstop) {
374                    argPos--;
375                }
376                // Start treating the rest as JSP Pages
377                break;
378            }
379        }
380 
381        // Add all extra arguments to the list of files
382        while( true ) {
383            String file = nextFile();
384            if( file==null ) {
385                break;
386            }
387            pages.add( file );
388        }
389    }
