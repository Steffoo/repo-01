689    protected static final String canonicalURI(String s) {
690       if (s == null) return null;
691       StringBuffer result = new StringBuffer();
692       final int len = s.length();
693       int pos = 0;
694       while (pos < len) {
695           char c = s.charAt(pos);
696           if ( isPathSeparator(c) ) {
697               /*
698                * multiple path separators.
699                * 'foo///bar' -> 'foo/bar'
700                */
701               while (pos+1 < len && isPathSeparator(s.charAt(pos+1))) {
702                   ++pos;
703               }
704 
705               if (pos+1 < len && s.charAt(pos+1) == '.') {
706                   /*
707                    * a single dot at the end of the path - we are done.
708                    */
709                   if (pos+2 >= len) break;
710 
711                   switch (s.charAt(pos+2)) {
712                       /*
713                        * self directory in path
714                        * foo/./bar -> foo/bar
715                        */
716                   case '/':
717                   case '\\':
718                       pos += 2;
719                       continue;
720 
721                       /*
722                        * two dots in a path: go back one hierarchy.
723                        * foo/bar/../baz -> foo/baz
724                        */
725                   case '.':
726                       // only if we have exactly _two_ dots.
727                       if (pos+3 < len && isPathSeparator(s.charAt(pos+3))) {
728                           pos += 3;
729                           int separatorPos = result.length()-1;
730                           while (separatorPos >= 0 && 
731                                  ! isPathSeparator(result
732                                                    .charAt(separatorPos))) {
733                               --separatorPos;
734                           }
735                           if (separatorPos >= 0)
736                               result.setLength(separatorPos);
737                           continue;
738                       }
739                   }
740               }
741           }
742           result.append(c);
743           ++pos;
744       }
745       return result.toString();
746    }
