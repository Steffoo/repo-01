480    /*
481     * Receives notification of character data inside an element.
482     *
483     * The SAX does not call this method with all of the template text, but may
484     * invoke this method with chunks of it.  This is a problem when we try
485     * to determine if the text contains only whitespaces, or when we are
486     * looking for an EL expression string.  Therefore it is necessary to
487     * buffer and concatenate the chunks and process the concatenated text
488     * later (at beginTag and endTag)
489     *
490     * @param buf The characters
491     * @param offset The start position in the character array
492     * @param len The number of characters to use from the character array
493     *
494     * @throws SAXException
495     */
496    public void characters(char[] buf, int offset, int len) {
497 
498        if (charBuffer == null) {
499            charBuffer = new StringBuffer();
500        }
501        charBuffer.append(buf, offset, len);
502    }
503 
504    private void processChars() throws SAXException {
505 
506        if (charBuffer == null || directivesOnly) {
507            return;
508        }
509 
510        /*
511         * JSP.6.1.1: All textual nodes that have only white space are to be
512         * dropped from the document, except for nodes in a jsp:text element,
513         * and any leading and trailing white-space-only textual nodes in a
514         * jsp:attribute whose 'trim' attribute is set to FALSE, which are to
515         * be kept verbatim.
516         * JSP.6.2.3 defines white space characters.
517         */
518        boolean isAllSpace = true;
519        if (!(current instanceof Node.JspText)
520            && !(current instanceof Node.NamedAttribute)) {
521            for (int i = 0; i < charBuffer.length(); i++) {
522                if (!(charBuffer.charAt(i) == ' '
523                    || charBuffer.charAt(i) == '\n'
524                    || charBuffer.charAt(i) == '\r'
525                    || charBuffer.charAt(i) == '\t')) {
526                    isAllSpace = false;
527                    break;
528                }
529            }
530        }
531 
532        if (!isAllSpace && tagDependentPending) {
533            tagDependentPending = false;
534            tagDependentNesting++;
535        }
536 
537        if (tagDependentNesting > 0) {
538            if (charBuffer.length() > 0) {
539                new Node.TemplateText(charBuffer.toString(), startMark, current);
540            }
541            startMark = new Mark(ctxt, path, locator.getLineNumber(),
542                                 locator.getColumnNumber());
543            charBuffer = null;
544            return;
545        }
546 
547        if ((current instanceof Node.JspText)
548            || (current instanceof Node.NamedAttribute)
549            || !isAllSpace) {
550 
551            int line = startMark.getLineNumber();
552            int column = startMark.getColumnNumber();
553 
554            CharArrayWriter ttext = new CharArrayWriter();
555            int lastCh = 0, elType = 0;
556            for (int i = 0; i < charBuffer.length(); i++) {
557 
558                int ch = charBuffer.charAt(i);
559                if (ch == '\n') {
560                    column = 1;
561                    line++;
562                } else {
563                    column++;
564                }
565                if ((lastCh == '$' || lastCh == '#') && ch == '{') {
566                    elType = lastCh;
567                    if (ttext.size() > 0) {
568                        new Node.TemplateText(
569                            ttext.toString(),
570                            startMark,
571                            current);
572                        ttext = new CharArrayWriter();
573                        //We subtract two from the column number to
574                        //account for the '[$,#]{' that we've already parsed
575                        startMark = new Mark(ctxt, path, line, column - 2);
576                    }
577                    // following "${" || "#{" to first unquoted "}"
578                    i++;
579                    boolean singleQ = false;
580                    boolean doubleQ = false;
581                    lastCh = 0;
582                    for (;; i++) {
583                        if (i >= charBuffer.length()) {
584                            throw new SAXParseException(
585                                Localizer.getMessage(
586                                    "jsp.error.unterminated",
587                                    (char) elType + "{"),
588                                locator);
589 
590                        }
591                        ch = charBuffer.charAt(i);
592                        if (ch == '\n') {
593                            column = 1;
594                            line++;
595                        } else {
596                            column++;
597                        }
598                        if (lastCh == '\\' && (singleQ || doubleQ)) {
599                            ttext.write(ch);
600                            lastCh = 0;
601                            continue;
602                        }
603                        if (ch == '}') {
604                            new Node.ELExpression((char) elType,
605                                ttext.toString(),
606                                startMark,
607                                current);
608                            ttext = new CharArrayWriter();
609                            startMark = new Mark(ctxt, path, line, column);
610                            break;
611                        }
612                        if (ch == '"')
613                            doubleQ = !doubleQ;
614                        else if (ch == '\'')
615                            singleQ = !singleQ;
616 
617                        ttext.write(ch);
618                        lastCh = ch;
619                    }
620                } else if (lastCh == '\\' && (ch == '$' || ch == '#')) {
621                    if (pageInfo.isELIgnored()) {
622                        ttext.write('\\');
623                    }
624                    ttext.write(ch);
625                    ch = 0;  // Not start of EL anymore
626                } else {
627                    if (lastCh == '$' || lastCh == '#' || lastCh == '\\') {
628                        ttext.write(lastCh);
629                    }
630                    if (ch != '$' && ch != '#' && ch != '\\') {
631                        ttext.write(ch);
632                    }
633                }
634                lastCh = ch;
635            }
636            if (lastCh == '$' || lastCh == '#' || lastCh == '\\') {
637                ttext.write(lastCh);
638            }
639            if (ttext.size() > 0) {
640                new Node.TemplateText(ttext.toString(), startMark, current);
641            }
642        }
643        startMark = new Mark(ctxt, path, locator.getLineNumber(),
644                             locator.getColumnNumber());
645 
646        charBuffer = null;
647    }

649    /*
650     * Receives notification of the end of an element.
651     */
652    public void endElement(String uri, String localName, String qName)
653        throws SAXException {
654 
655        processChars();
656 
657        if (directivesOnly &&
658            !(JSP_URI.equals(uri) && localName.startsWith(DIRECTIVE_ACTION))) {
659            return;
660        }
661 
662        if (current instanceof Node.NamedAttribute) {
663            boolean isTrim = ((Node.NamedAttribute)current).isTrim();
664            Node.Nodes subElems = ((Node.NamedAttribute)current).getBody();
665            for (int i = 0; subElems != null && i < subElems.size(); i++) {
666                Node subElem = subElems.getNode(i);
667                if (!(subElem instanceof Node.TemplateText)) {
668                    continue;
669                }
670                // Ignore any whitespace (including spaces, carriage returns,
671                // line feeds, and tabs, that appear at the beginning and at
672                // the end of the body of the <jsp:attribute> action, if the
673                // action's 'trim' attribute is set to TRUE (default).
674                // In addition, any textual nodes in the <jsp:attribute> that
675                // have only white space are dropped from the document, with
676                // the exception of leading and trailing white-space-only
677                // textual nodes in a <jsp:attribute> whose 'trim' attribute
678                // is set to FALSE, which must be kept verbatim.
679                if (i == 0) {
680                    if (isTrim) {
681                        ((Node.TemplateText)subElem).ltrim();
682                    }
683                } else if (i == subElems.size() - 1) {
684                    if (isTrim) {
685                        ((Node.TemplateText)subElem).rtrim();
686                    }
687                } else {
688                    if (((Node.TemplateText)subElem).isAllSpace()) {
689                        subElems.remove(subElem);
690                    }
691                }
692            }
693        } else if (current instanceof Node.ScriptingElement) {
694            checkScriptingBody((Node.ScriptingElement)current);
695        }
696 
697        if ( isTagDependent(current)) {
698            tagDependentNesting--;
699        }
700 
701        if (scriptlessBodyNode != null
702                && current.equals(scriptlessBodyNode)) {
703            scriptlessBodyNode = null;
704        }
705 
706        if (current instanceof Node.CustomTag) {
707                String bodyType = getBodyType((Node.CustomTag) current);
708                if (TagInfo.BODY_CONTENT_EMPTY.equalsIgnoreCase(bodyType)) {
709                        // Children - if any - must be JSP attributes
710                        Node.Nodes children = current.getBody();
711                        if (children != null && children.size() > 0) {
712                                for (int i = 0; i < children.size(); i++) {
713                                        Node child = children.getNode(i);
714                                        if (!(child instanceof Node.NamedAttribute)) {
715                                                throw new SAXParseException(Localizer.getMessage(
716                                                                "jasper.error.emptybodycontent.nonempty",
717                                                                current.qName), locator);
718                                        }
719                                }
720                        }
721                }
722        }
723        if (current.getParent() != null) {
724            current = current.getParent();
725        }
726    }
