845         * was created from a JSP document, its text was stored as one or more

850        public String getText() {
851            String ret = text;
852            if (ret == null) {
853                if (body != null) {
854                    StringBuffer buf = new StringBuffer();
855                    for (int i = 0; i < body.size(); i++) {
856                        buf.append(body.getNode(i).getText());
857                    }
858                    ret = buf.toString();
859                } else {
860                    // Nulls cause NPEs further down the line
861                    ret = "";
862                }
863            }
864            return ret;
865        }

872            if (text == null && body != null && body.size() > 0) {
