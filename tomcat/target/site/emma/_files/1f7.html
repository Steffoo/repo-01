431         * The value of the following attributes must be unique. 1. 'name'
432         * attribute of an attribute directive 2. 'name-given' attribute of a
433         * variable directive 3. 'alias' attribute of variable directive 4.
434         * 'dynamic-attributes' of a tag directive except that
435         * 'dynamic-attributes' can (and must) have the same value when it
436         * appears in multiple tag directives.
437         *
438         * Also, 'name-from' attribute of a variable directive cannot have the
439         * same value as that from another variable directive.
440         */
441        private void checkUniqueName(String name, String type, Node n)
442                throws JasperException {
443            checkUniqueName(name, type, n, null);
444        }

446        private void checkUniqueName(String name, String type, Node n,
447                TagAttributeInfo attr) throws JasperException {
448 
449            HashMap table = (type == VAR_NAME_FROM) ? nameFromTable : nameTable;
450            NameEntry nameEntry = (NameEntry) table.get(name);
451            if (nameEntry != null) {
452                if (type != TAG_DYNAMIC || nameEntry.getType() != TAG_DYNAMIC) {
453                    int line = nameEntry.getNode().getStart().getLineNumber();
454                    err.jspError(n, "jsp.error.tagfile.nameNotUnique", type,
455                            nameEntry.getType(), Integer.toString(line));
456                }
457            } else {
458                table.put(name, new NameEntry(type, n, attr));
459            }

482                            || tagAttr.canBeRequestTime()) {
483                        err.jspError(nameFromNode,
484                                "jsp.error.tagfile.nameFrom.badAttribute",
485                                nameFrom, Integer.toString(node.getStart()
486                                        .getLineNumber()));
487                    }

561    /**
562     * Compiles and loads a tagfile.
563     */
564    private Class loadTagFile(Compiler compiler, String tagFilePath,
565            TagInfo tagInfo, PageInfo parentPageInfo) throws JasperException {
566 
567        URL tagFileJarUrl = null;
568        if (tagFilePath.startsWith("/META-INF/")) {
569            try {
570                tagFileJarUrl = new URL("jar:" +
571                        compiler.getCompilationContext().getTldLocation(
572                        tagInfo.getTagLibrary().getURI())[0] + "!/");
573            } catch (MalformedURLException e) {
574                // Ignore - tagFileJarUrl will be null
575            }
576        }
577        String tagFileJarPath;
578        if (tagFileJarUrl == null) {
579            tagFileJarPath = "";
580        } else {
581            tagFileJarPath = tagFileJarUrl.toString();
582        }
583 
584        JspCompilationContext ctxt = compiler.getCompilationContext();
585        JspRuntimeContext rctxt = ctxt.getRuntimeContext();
586        String wrapperUri = tagFileJarPath + tagFilePath;
587        JspServletWrapper wrapper = rctxt.getWrapper(wrapperUri);
588 
589        synchronized (rctxt) {
590            if (wrapper == null) {
591                wrapper = new JspServletWrapper(ctxt.getServletContext(), ctxt
592                        .getOptions(), tagFilePath, tagInfo, ctxt
593                        .getRuntimeContext(), tagFileJarUrl);
594                rctxt.addWrapper(wrapperUri, wrapper);
595 
596                // Use same classloader and classpath for compiling tag files
597                wrapper.getJspEngineContext().setClassLoader(
598                        (URLClassLoader) ctxt.getClassLoader());
599                wrapper.getJspEngineContext().setClassPath(ctxt.getClassPath());
600            } else {
601                // Make sure that JspCompilationContext gets the latest TagInfo
602                // for the tag file. TagInfo instance was created the last
603                // time the tag file was scanned for directives, and the tag
604                // file may have been modified since then.
605                wrapper.getJspEngineContext().setTagInfo(tagInfo);
606            }
607 
608            Class tagClazz;
609            int tripCount = wrapper.incTripCount();
610            try {
611                if (tripCount > 0) {
612                    // When tripCount is greater than zero, a circular
613                    // dependency exists. The circularily dependant tag
614                    // file is compiled in prototype mode, to avoid infinite
615                    // recursion.
616 
617                    JspServletWrapper tempWrapper = new JspServletWrapper(ctxt
618                            .getServletContext(), ctxt.getOptions(),
619                            tagFilePath, tagInfo, ctxt.getRuntimeContext(),
620                            tagFileJarUrl);
621                    // Use same classloader and classpath for compiling tag files
622                    tempWrapper.getJspEngineContext().setClassLoader(
623                            (URLClassLoader) ctxt.getClassLoader());
624                    tempWrapper.getJspEngineContext().setClassPath(ctxt.getClassPath());
625                    tagClazz = tempWrapper.loadTagFilePrototype();
626                    tempVector.add(tempWrapper.getJspEngineContext()
627                            .getCompiler());
628                } else {
629                    tagClazz = wrapper.loadTagFile();
630                }
631            } finally {
632                wrapper.decTripCount();
633            }
634 
635            // Add the dependants for this tag file to its parent's
636            // dependant list. The only reliable dependency information
637            // can only be obtained from the tag instance.
638            try {
639                Object tagIns = tagClazz.newInstance();
640                if (tagIns instanceof JspSourceDependent) {
641                    Iterator iter = ((List) ((JspSourceDependent) tagIns)
642                            .getDependants()).iterator();
643                    while (iter.hasNext()) {
644                        parentPageInfo.addDependant((String) iter.next());
645                    }
646                }
647            } catch (Exception e) {
648                // ignore errors
649            }
650 
651            return tagClazz;
652        }
653    }

655    /*
656     * Visitor which scans the page and looks for tag handlers that are tag
657     * files, compiling (if necessary) and loading them.
658     */
659    private class TagFileLoaderVisitor extends Node.Visitor {
660 
661        private Compiler compiler;
662 
663        private PageInfo pageInfo;
664 
665        TagFileLoaderVisitor(Compiler compiler) {
666 
667            this.compiler = compiler;
668            this.pageInfo = compiler.getPageInfo();
669        }

671        public void visit(Node.CustomTag n) throws JasperException {
672            TagFileInfo tagFileInfo = n.getTagFileInfo();
673            if (tagFileInfo != null) {
674                String tagFilePath = tagFileInfo.getPath();
675                if (tagFilePath.startsWith("/META-INF/")) {
676                    // For tags in JARs, add the TLD and the tag as a dependency
677                    String[] location =
678                        compiler.getCompilationContext().getTldLocation(
679                            tagFileInfo.getTagInfo().getTagLibrary().getURI());
680                    // Add TLD
681                    pageInfo.addDependant("jar:" + location[0] + "!/" +
682                            location[1]);
683                    // Add Tag
684                    pageInfo.addDependant("jar:" + location[0] + "!" +
685                            tagFilePath);
686                } else {
687                    pageInfo.addDependant(tagFilePath);
688                }
689                Class c = loadTagFile(compiler, tagFilePath, n.getTagInfo(),
690                        pageInfo);
691                n.setTagHandlerClass(c);
692            }
693            visitBody(n);
694        }
695    }
