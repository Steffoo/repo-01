486            } else if ("icon".equals(tname)
487                    || "display-name".equals(tname)
488                    || "description".equals(tname)) {
489                // Ignore these elements: Bugzilla 38015
490            } else {
491                if (log.isWarnEnabled()) {
492                    log.warn(Localizer.getMessage(
493                            "jsp.warning.unknown.element.in.tagfile", tname));

502            // it has been removed

504        } else if (!path.startsWith("/WEB-INF/tags")) {

506        }
507

509                parserController, name, path, jarFileUrl, this);
510        return new TagFileInfo(name, path, tagInfo);
511    }

513    TagAttributeInfo createAttribute(TreeNode elem, String jspVersion) {
514        String name = null;
515        String type = null;
516        String expectedType = null;
517        String methodSignature = null;
518        boolean required = false, rtexprvalue = false, reqTime = false, isFragment = false, deferredValue = false, deferredMethod = false;
519 
520        Iterator list = elem.findChildren();
521        while (list.hasNext()) {
522            TreeNode element = (TreeNode) list.next();
523            String tname = element.getName();
524 
525            if ("name".equals(tname)) {
526                name = element.getBody();
527            } else if ("required".equals(tname)) {
528                String s = element.getBody();
529                if (s != null)
530                    required = JspUtil.booleanValue(s);
531            } else if ("rtexprvalue".equals(tname)) {
532                String s = element.getBody();
533                if (s != null)
534                    rtexprvalue = JspUtil.booleanValue(s);
535            } else if ("type".equals(tname)) {
536                type = element.getBody();
537                if ("1.2".equals(jspVersion)
538                        && (type.equals("Boolean") || type.equals("Byte")
539                                || type.equals("Character")
540                                || type.equals("Double")
541                                || type.equals("Float")
542                                || type.equals("Integer")
543                                || type.equals("Long") || type.equals("Object")
544                                || type.equals("Short") || type
545                                .equals("String"))) {
546                    type = "java.lang." + type;
547                }
548            } else if ("fragment".equals(tname)) {
549                String s = element.getBody();
550                if (s != null) {
551                    isFragment = JspUtil.booleanValue(s);
552                }
553            } else if ("deferred-value".equals(tname)) {
554                deferredValue = true;
555                type = "javax.el.ValueExpression";
556                TreeNode child = element.findChild("type");
557                if (child != null) {
558                    expectedType = child.getBody();
559                    if (expectedType != null) {
560                        expectedType = expectedType.trim();
561                    }
562                } else {
563                    expectedType = "java.lang.Object";
564                }
565            } else if ("deferred-method".equals(tname)) {
566                deferredMethod = true;
567                type = "javax.el.MethodExpression";
568                TreeNode child = element.findChild("method-signature");
569                if (child != null) {
570                    methodSignature = child.getBody();
571                    if (methodSignature != null) {
572                        methodSignature = methodSignature.trim();
573                    }
574                } else {
575                    methodSignature = "java.lang.Object method()";
576                }
577            } else if ("description".equals(tname) || // Ignored elements
578            false) {
579                ;
580            } else {
581                if (log.isWarnEnabled()) {
582                    log.warn(Localizer.getMessage(
583                            "jsp.warning.unknown.element.in.attribute", tname));
584                }
585            }
586        }
587 
588        if (isFragment) {
589            /*
590             * According to JSP.C-3 ("TLD Schema Element Structure - tag"),
591             * 'type' and 'rtexprvalue' must not be specified if 'fragment' has
592             * been specified (this will be enforced by validating parser).
593             * Also, if 'fragment' is TRUE, 'type' is fixed at
594             * javax.servlet.jsp.tagext.JspFragment, and 'rtexprvalue' is fixed
595             * at true. See also JSP.8.5.2.
596             */
597            type = "javax.servlet.jsp.tagext.JspFragment";
598            rtexprvalue = true;
599        }
600 
601        if (!rtexprvalue && type == null) {
602            // According to JSP spec, for static values (those determined at
603            // translation time) the type is fixed at java.lang.String.
604            type = "java.lang.String";
605        }
606 
607        return new TagAttributeInfo(name, required, type, rtexprvalue,
608                isFragment, null, deferredValue, deferredMethod, expectedType,
609                methodSignature);
610    }
