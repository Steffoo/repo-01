442            Iterator it = dirList.iterator();
443            while (it.hasNext()) {
444                String path = (String) it.next();
445                if (path.endsWith("/")) {
446                    processTldsInFileSystem(path);
447                }
448                if (!path.endsWith(".tld")) {
449                    continue;
450                }
451                InputStream stream = ctxt.getResourceAsStream(path);
452                String uri = null;
453                try {
454                    uri = getUriFromTld(path, stream);
455                } finally {
456                    if (stream != null) {
457                        try {
458                            stream.close();
459                        } catch (Throwable t) {
460                            // do nothing
461                        }
462                    }
463                }
464                // Add implicit map entry only if its uri is not already
465                // present in the map
466                if (uri != null && mappings.get(uri) == null) {
467                    mappings.put(uri, new String[] { path, null });
468                }
469            }
470        }
471    }
472 
473    /*
474     * Returns the value of the uri element of the given TLD, or null if the
475     * given TLD does not contain any such element.
476     */
477    private String getUriFromTld(String resourcePath, InputStream in)
478        throws JasperException
479    {
480        boolean validate = Boolean.parseBoolean(
481                ctxt.getInitParameter(
482                        Constants.XML_VALIDATION_TLD_INIT_PARAM));
483        String blockExternalString = ctxt.getInitParameter(
484                Constants.XML_BLOCK_EXTERNAL_INIT_PARAM);
485        boolean blockExternal;
486        if (blockExternalString == null) {
487            blockExternal = true;
488        } else {
489            blockExternal = Boolean.parseBoolean(blockExternalString);
490        }
491 
492        ParserUtils pu = new ParserUtils(validate, blockExternal);
493        TreeNode tld = pu.parseXMLDocument(resourcePath, in);
494        TreeNode uri = tld.findChild("uri");
495        if (uri != null) {
496            String body = uri.getBody();
497            if (body != null)
498                return body;
499        }
500 
501        return null;
502    }

504    /*
505     * Scans all JARs accessible to the webapp's classloader and its
506     * parent classloaders for TLDs.
507     *
508     * The list of JARs always includes the JARs under WEB-INF/lib, as well as
509     * all shared JARs in the classloader delegation chain of the webapp's
510     * classloader.
511     *
512     * Considering JARs in the classloader delegation chain constitutes a
513     * Tomcat-specific extension to the TLD search
514     * order defined in the JSP spec. It allows tag libraries packaged as JAR
515     * files to be shared by web applications by simply dropping them in a
516     * location that all web applications have access to (e.g.,
517     * <CATALINA_HOME>/common/lib).
518     *
519     * The set of shared JARs to be scanned for TLDs is narrowed down by
520     * the <tt>noTldJars</tt> class variable, which contains the names of JARs
521     * that are known not to contain any TLDs.
522     */
523    private void scanJars() throws Exception {
524 
525        ClassLoader webappLoader
526            = Thread.currentThread().getContextClassLoader();
527        ClassLoader loader = webappLoader;
528 
529        while (loader != null) {
530            if (loader instanceof URLClassLoader) {
531                URL[] urls = ((URLClassLoader) loader).getURLs();
532                for (int i=0; i<urls.length; i++) {
533                    URLConnection conn = urls[i].openConnection();
534                    if (conn instanceof JarURLConnection) {
535                        if (needScanJar(loader, webappLoader,
536                                        ((JarURLConnection) conn).getJarFile().getName())) {
537                            scanJar((JarURLConnection) conn, true);
538                        }
539                    } else {
540                        String urlStr = urls[i].toString();
541                        if (urlStr.startsWith(FILE_PROTOCOL)
542                                && urlStr.endsWith(JAR_FILE_SUFFIX)
543                                && needScanJar(loader, webappLoader, urlStr)) {
544                            URL jarURL = new URL("jar:" + urlStr + "!/");
545                            scanJar((JarURLConnection) jarURL.openConnection(),
546                                    true);
547                        }
548                    }
549                }
550            }
551 
552            loader = loader.getParent();
553        }
554    }
555 
556    /*
557     * Determines if the JAR file with the given <tt>jarPath</tt> needs to be
558     * scanned for TLDs.
559     *
560     * @param loader The current classloader in the parent chain
561     * @param webappLoader The webapp classloader
562     * @param jarPath The JAR file path
563     *
564     * @return TRUE if the JAR file identified by <tt>jarPath</tt> needs to be
565     * scanned for TLDs, FALSE otherwise
566     */
567    private boolean needScanJar(ClassLoader loader, ClassLoader webappLoader,
568                                String jarPath) {
569        if (loader == webappLoader) {
570            // JARs under WEB-INF/lib must be scanned unconditionally according
571            // to the spec.
572            return true;
573        } else {
574            String jarName = jarPath;
575            int slash = jarPath.lastIndexOf('/');
576            if (slash >= 0) {
577                jarName = jarPath.substring(slash + 1);
578            }
579            return (!noTldJars.contains(jarName));
580        }
581    }
