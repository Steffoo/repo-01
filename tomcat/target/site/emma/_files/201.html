654    /**
655     * Initialize the endpoint.
656     */
657    public void init()
658        throws Exception {
659 
660        if (initialized)
661            return;
662 
663        if (rootPool != 0) {
664            // A previous init() call failed
665            throw new IllegalStateException(
666                    sm.getString("endpoint.apr.previousInitFailed"));
667        }
668 
669        // Create the root APR memory pool
670        rootPool = Pool.create(0);
671        // Create the pool for the server socket
672        serverSockPool = Pool.create(rootPool);
673        // Create the APR address that will be bound
674        String addressStr = null;
675        if (address == null) {
676            addressStr = null;
677        } else {
678            addressStr = address.getHostAddress();
679        }
680        int family = Socket.APR_INET;
681        if (Library.APR_HAVE_IPV6) {
682            if (addressStr == null) {
683                if (!OS.IS_BSD && !OS.IS_WIN32 && !OS.IS_WIN64)
684                    family = Socket.APR_UNSPEC;
685            } else if (addressStr.indexOf(':') >= 0) {
686                family = Socket.APR_UNSPEC;
687            }
688         }
689 
690        long inetAddress = Address.info(addressStr, family,
691                port, 0, rootPool);
692        // Create the APR server socket
693        serverSock = Socket.create(Address.getInfo(inetAddress).family,
694                Socket.SOCK_STREAM,
695                Socket.APR_PROTO_TCP, rootPool);
696        if (OS.IS_UNIX) {
697            Socket.optSet(serverSock, Socket.APR_SO_REUSEADDR, 1);
698        }
699        // Deal with the firewalls that tend to drop the inactive sockets
700        Socket.optSet(serverSock, Socket.APR_SO_KEEPALIVE, 1);
701        // Bind the server socket
