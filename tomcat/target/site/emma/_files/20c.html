397            return written;
398        } else {
399            //are we closing or closed?
400            if ( closing || closed) throw new IOException("Channel is in closing state.");
401 
402            //the number of bytes written
403            int written = 0;
404 
405            if (!flush(netOutBuffer)) {
406                //we haven't emptied out the buffer yet
407                return written;
408            }
409 
410            /*
411             * The data buffer is empty, we can reuse the entire buffer.
412             */
413            netOutBuffer.clear();
414 
415            SSLEngineResult result = sslEngine.wrap(src, netOutBuffer);
416            written = result.bytesConsumed();
417            netOutBuffer.flip();
418 
419            if (result.getStatus() == Status.OK) {
420                if (result.getHandshakeStatus() == HandshakeStatus.NEED_TASK) tasks();
421            } else {
422                throw new IOException("Unable to wrap data, invalid engine state: " +result.getStatus());
423            }        
424 
425            //force a flush
426            flush(netOutBuffer);
427 
428            return written;
429        }
430    }
