546    /*
547     * Scans the web application's subdirectory identified by rootPath,
548     * along with its subdirectories, for TLDs.
549     *
550     * Initially, rootPath equals /WEB-INF. The /WEB-INF/classes and
551     * /WEB-INF/lib subdirectories are excluded from the search, as per the
552     * JSP 2.0 spec.
553     *
554     * @param resources The web application's resources
555     * @param rootPath The path whose subdirectories are to be searched for
556     * TLDs
557     * @param tldPaths The set of TLD resource paths to add to
558     */
559    private void tldScanResourcePathsWebInf(DirContext resources,
560                                            String rootPath,
561                                            Set tldPaths)
562            throws IOException {
563 
564        if (log.isTraceEnabled()) {
565            log.trace("  Scanning TLDs in " + rootPath + " subdirectory");
566        }
567 
568        try {
569            NamingEnumeration items = resources.list(rootPath);
570            while (items.hasMoreElements()) {
571                NameClassPair item = (NameClassPair) items.nextElement();
572                String resourcePath = rootPath + "/" + item.getName();
573                if (!resourcePath.endsWith(".tld")
574                        && (resourcePath.startsWith("/WEB-INF/classes")
575                            || resourcePath.startsWith("/WEB-INF/lib"))) {
576                    continue;
577                }
578                if (resourcePath.endsWith(".tld")) {
579                    if (log.isTraceEnabled()) {
580                        log.trace("   Adding path '" + resourcePath + "'");
581                    }
582                    tldPaths.add(resourcePath);
583                } else {
584                    tldScanResourcePathsWebInf(resources, resourcePath,
585                                               tldPaths);
586                }
587            }
588        } catch (NamingException e) {
589            ; // Silent catch: it's valid that no /WEB-INF directory exists
590        }
591    }

593    /**
594     * Returns a map of the paths to all JAR files that are accessible to the
595     * webapp and will be scanned for TLDs.
596     *
597     * The map always includes all the JARs under WEB-INF/lib, as well as
598     * shared JARs in the classloader delegation chain of the webapp's
599     * classloader.
600     *
601     * The latter constitutes a Tomcat-specific extension to the TLD search
602     * order defined in the JSP spec. It allows tag libraries packaged as JAR
603     * files to be shared by web applications by simply dropping them in a
604     * location that all web applications have access to (e.g.,
605     * <CATALINA_HOME>/common/lib).
606     *
607     * The set of shared JARs to be scanned for TLDs is narrowed down by
608     * the <tt>noTldJars</tt> class variable, which contains the names of JARs
609     * that are known not to contain any TLDs.
610     *
611     * @return Map of JAR file paths
612     */
613    private Map getJarPaths() {
614 
615        HashMap jarPathMap = null;
616 
617        ClassLoader webappLoader = Thread.currentThread().getContextClassLoader();
618        ClassLoader loader = webappLoader;
619        while (loader != null) {
620            if (loader instanceof URLClassLoader) {
621                URL[] urls = ((URLClassLoader) loader).getURLs();
622                for (int i=0; i<urls.length; i++) {
623                    // Expect file URLs, these are %xx encoded or not depending
624                    // on the class loader
625                    // This is definitely not as clean as using JAR URLs either
626                    // over file or the custom jndi handler, but a lot less
627                    // buggy overall
628 
629                    // Check that the URL is using file protocol, else ignore it
630                    if (!"file".equals(urls[i].getProtocol())) {
631                        continue;
632                    }
633 
634                    File file = null;
635                    try {
636                        file = new File(urls[i].toURI());
637                    } catch (URISyntaxException e) {
638                        // Ignore, probably an unencoded char
639                        file = new File(urls[i].getFile());
640                    }
641                    try {
642                        file = file.getCanonicalFile();
643                    } catch (IOException e) {
644                        // Ignore
645                    }
646                    if (!file.exists()) {
647                        continue;
648                    }
649                    String path = file.getAbsolutePath();
650                    if (!path.endsWith(".jar")) {
651                        continue;
652                    }
653                    /*
654                     * Scan all JARs from WEB-INF/lib, plus any shared JARs
655                     * that are not known not to contain any TLDs
656                     */
657                    if (loader == webappLoader
658                            || noTldJars == null
659                            || !noTldJars.contains(file.getName())) {
660                        if (jarPathMap == null) {
661                            jarPathMap = new HashMap();
662                            jarPathMap.put(path, file);
663                        } else if (!jarPathMap.containsKey(path)) {
664                            jarPathMap.put(path, file);
665                        }
666                    }
667                }
668            }
669            loader = loader.getParent();
670        }
671 
672        return jarPathMap;
673    }
