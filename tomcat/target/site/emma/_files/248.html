26/**
27 * <p>Rule implementation that calls a method on an object on the stack
28 * (normally the top/parent object), passing arguments collected from 
29 * subsequent <code>CallParamRule</code> rules or from the body of this
30 * element. </p>
31 *
32 * <p>By using {@link #CallMethodRule(String methodName)} 
33 * a method call can be made to a method which accepts no
34 * arguments.</p>
35 *
36 * <p>Incompatible method parameter types are converted 
37 * using <code>org.apache.commons.beanutils.ConvertUtils</code>.
38 * </p>
39 *
40 * <p>This rule now uses
41 * <a href="http://commons.apache.org/beanutils/apidocs/org/apache/commons/beanutils/MethodUtils.html">
42 * org.apache.commons.beanutils.MethodUtils#invokeMethod
43 * </a> by default.
44 * This increases the kinds of methods successfully and allows primitives
45 * to be matched by passing in wrapper classes.
46 * There are rare cases when org.apache.commons.beanutils.MethodUtils#invokeExactMethod 
47 * (the old default) is required.
48 * This method is much stricter in its reflection.
49 * Setting the <code>UseExactMatch</code> to true reverts to the use of this 
50 * method.</p>
51 *
52 * <p>Note that the target method is invoked when the  <i>end</i> of
53 * the tag the CallMethodRule fired on is encountered, <i>not</i> when the
54 * last parameter becomes available. This implies that rules which fire on
55 * tags nested within the one associated with the CallMethodRule will 
56 * fire before the CallMethodRule invokes the target method. This behaviour is
57 * not configurable. </p>
58 *
59 * <p>Note also that if a CallMethodRule is expecting exactly one parameter
60 * and that parameter is not available (eg CallParamRule is used with an
61 * attribute name but the attribute does not exist) then the method will
62 * not be invoked. If a CallMethodRule is expecting more than one parameter,
63 * then it is always invoked, regardless of whether the parameters were

467     * Process the end of this element.
468     */
469    public void end() throws Exception {
470 
471        // Retrieve or construct the parameter values array
472        Object parameters[] = null;
473        if (paramCount > 0) {
474 
475            parameters = (Object[]) digester.popParams();
476            
477            if (digester.log.isTraceEnabled()) {
478                for (int i=0,size=parameters.length;i<size;i++) {
479                    digester.log.trace("[CallMethodRule](" + i + ")" + parameters[i]) ;
480                }
481            }
482            
483            // In the case where the parameter for the method
484            // is taken from an attribute, and that attribute
485            // isn't actually defined in the source XML file,
486            // skip the method call
487            if (paramCount == 1 && parameters[0] == null) {
488                return;
489            }
490 
491        } else if (paramTypes != null && paramTypes.length != 0) {
492 
493            // In the case where the parameter for the method
494            // is taken from the body text, but there is no
495            // body text included in the source XML file,
496            // skip the method call
497            if (bodyText == null) {
498                return;
499            }
500 
501            parameters = new Object[1];
502            parameters[0] = bodyText;
503            if (paramTypes.length == 0) {
504                paramTypes = new Class[1];
505                paramTypes[0] = "abc".getClass();
506            }
507 
508        }
509 
510        // Construct the parameter values array we will need
511        // We only do the conversion if the param value is a String and
512        // the specified paramType is not String. 
513        Object paramValues[] = new Object[paramTypes.length];
514        for (int i = 0; i < paramTypes.length; i++) {
515            // convert nulls and convert stringy parameters 
516            // for non-stringy param types
517            if(
518                parameters[i] == null ||
519                 (parameters[i] instanceof String && 
520                   !String.class.isAssignableFrom(paramTypes[i]))) {
521                
522                paramValues[i] =
523                        IntrospectionUtils.convert((String) parameters[i], paramTypes[i]);
524            } else {
525                paramValues[i] = parameters[i];
526            }
527        }
528 
529        // Determine the target object for the method call
530        Object target;
531        if (targetOffset >= 0) {
532            target = digester.peek(targetOffset);
533        } else {
534            target = digester.peek( digester.getCount() + targetOffset );
535        }
536        
537        if (target == null) {
538            StringBuffer sb = new StringBuffer();
539            sb.append("[CallMethodRule]{");
540            sb.append(digester.match);
541            sb.append("} Call target is null (");
542            sb.append("targetOffset=");
543            sb.append(targetOffset);
544            sb.append(",stackdepth=");
545            sb.append(digester.getCount());
546            sb.append(")");
547            throw new org.xml.sax.SAXException(sb.toString());
548        }
549        
550        // Invoke the required method on the top object
551        if (digester.log.isDebugEnabled()) {
552            StringBuffer sb = new StringBuffer("[CallMethodRule]{");
553            sb.append(digester.match);
554            sb.append("} Call ");
555            sb.append(target.getClass().getName());
556            sb.append(".");
557            sb.append(methodName);
558            sb.append("(");
559            for (int i = 0; i < paramValues.length; i++) {
560                if (i > 0) {
561                    sb.append(",");
562                }
563                if (paramValues[i] == null) {
564                    sb.append("null");
565                } else {
566                    sb.append(paramValues[i].toString());
567                }
568                sb.append("/");
569                if (paramTypes[i] == null) {
570                    sb.append("null");
571                } else {
572                    sb.append(paramTypes[i].getName());
573                }
574            }
575            sb.append(")");
576            digester.log.debug(sb.toString());
577        }
578        Object result = IntrospectionUtils.callMethodN(target, methodName,
579                paramValues, paramTypes);   
580        processMethodCallResult(result);
581    }
