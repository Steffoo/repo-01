167     *  @param value - The value to wrap
168     *  @return '-' if null. Otherwise, toString() will
169     *     be called on the object and the value will be wrapped
170     *     in quotes and any quotes will be escaped with 2
171     *     sets of quotes.
172     */
173    static String wrap(Object value) {
174        String svalue;
175        // Does the value contain a " ? If so must encode it
176        if (value == null || "-".equals(value))
177            return "-";
178 
179        try {
180            svalue = value.toString();
181        } catch (Throwable e) {
182            /* Log error */
183            return "-";
184        }
185 
186        /* Wrap all values in double quotes. */
187        StringBuffer buffer = new StringBuffer(svalue.length() + 2);
188        buffer.append('\"');
189        int i = 0;
190        while (i < svalue.length()) {
191            int j = svalue.indexOf('\"', i);
192            if (j == -1) {
193                buffer.append(svalue.substring(i));
194                i = svalue.length();
195            } else {
196                buffer.append(svalue.substring(i, j + 1));
197                buffer.append('"');
198                i = j + 1;
199            }
200        }
201 
202        buffer.append('\"');
203        return buffer.toString();
204    }

411    protected class PatternTokenizer {
412        private StringReader sr = null;
413        private StringBuffer buf = new StringBuffer();
414        private boolean ended = false;
415        private boolean subToken;
416        private boolean parameter;
417        
418        public PatternTokenizer(String str) {
419            sr = new StringReader(str);
420        }
421        
422        public boolean hasSubToken() {
423            return subToken;
424        }

426        public boolean hasParameter() {
427            return parameter;
428        }

430        public String getToken() throws IOException {
431            if(ended)
432                return null ;
433            
434            String result = null;
435            subToken = false;
436            parameter = false;
437            
438            int c = sr.read();
439            while (c != -1) {
440                switch (c) {
441                case ' ':
442                    result = buf.toString();
443                    buf = new StringBuffer();
444                    buf.append((char) c);
445                    return result;
446                case '-':
447                    result = buf.toString();
448                    buf = new StringBuffer();
449                    subToken = true;
450                    return result;
451                case '(':
452                    result = buf.toString();
453                    buf = new StringBuffer();
454                    parameter = true;
455                    return result;
456                case ')':
457                    result = buf.toString();
458                    buf = new StringBuffer();
459                    break;
460                default:
461                    buf.append((char) c);
462                }
463                c = sr.read();
464            }
465            ended = true;
466            if (buf.length() != 0) {
467                return buf.toString();
468            } else {
469                return null;
470            }
471        }

473        public String getParameter()throws IOException {
474            String result;
475            if (!parameter) {
476                return null;
477            }
478            parameter = false;
479            int c = sr.read();
480            while (c != -1) {
481                if (c == ')') {
482                    result = buf.toString();
483                    buf = new StringBuffer();
484                    return result;
485                }
486                buf.append((char) c);
487                c = sr.read();
488            }
489            return null;
490        }

492        public String getWhiteSpaces() throws IOException {
493            if(isEnded())
494                return "" ;
495            StringBuffer whiteSpaces = new StringBuffer();
496            if (buf.length() > 0) {
497                whiteSpaces.append(buf);
498                buf = new StringBuffer();
499            }
500            int c = sr.read();
501            while (Character.isWhitespace((char) c)) {
502                whiteSpaces.append((char) c);
503                c = sr.read();
504            }
505            if (c == -1) {
506                ended = true;
507            } else {
508                buf.append((char) c);
509            }
510            return whiteSpaces.toString();
511        }

514            return ended;
515        }

529            log.debug("decodePattern, pattern =" + pattern);
530        }
531        List<AccessLogElement> list = new ArrayList<AccessLogElement>();
532 
533        PatternTokenizer tokenizer = new PatternTokenizer(pattern);
534        try {
535 
536            // Ignore leading whitespace.
537            tokenizer.getWhiteSpaces();
538 
539            if (tokenizer.isEnded()) {
540                log.info("pattern was just empty or whitespace");
541                return null;

545            while (token != null) {
