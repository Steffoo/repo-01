406     * Convert an array of strings in a comma delimited string
407     */
408    protected static String listToCommaDelimitedString(List<String> stringList) {
409        if (stringList == null) {
410            return "";
411        }
412        StringBuilder result = new StringBuilder();
413        for (Iterator<String> it = stringList.iterator(); it.hasNext();) {
414            Object element = it.next();
415            if (element != null) {
416                result.append(element);
417                if (it.hasNext()) {
418                    result.append(", ");
419                }
420            }
421        }
422        return result.toString();

552    /**
553     * {@inheritDoc}
554     */
555    @Override
556    public void invoke(Request request, Response response) throws IOException, ServletException {
557        final String originalRemoteAddr = request.getRemoteAddr();
558        final String originalRemoteHost = request.getRemoteHost();
559        final String originalScheme = request.getScheme();
560        final boolean originalSecure = request.isSecure();
561        final int originalServerPort = request.getServerPort();
562        
563        if (matchesOne(originalRemoteAddr, internalProxies)) {
564            String remoteIp = null;
565            // In java 6, proxiesHeaderValue should be declared as a java.util.Deque
566            LinkedList<String> proxiesHeaderValue = new LinkedList<String>();
567            StringBuffer concatRemoteIpHeaderValue = new StringBuffer();
568            
569            for (Enumeration<String> e = request.getHeaders(remoteIpHeader); e.hasMoreElements();) {
570                if (concatRemoteIpHeaderValue.length() > 0) {
571                    concatRemoteIpHeaderValue.append(", ");
572                }
573 
574                concatRemoteIpHeaderValue.append(e.nextElement());
575            }
576 
577            String[] remoteIpHeaderValue = commaDelimitedListToStringArray(concatRemoteIpHeaderValue.toString());
578            int idx;
579            // loop on remoteIpHeaderValue to find the first trusted remote ip and to build the proxies chain
580            for (idx = remoteIpHeaderValue.length - 1; idx >= 0; idx--) {
581                String currentRemoteIp = remoteIpHeaderValue[idx];
582                remoteIp = currentRemoteIp;
583                if (matchesOne(currentRemoteIp, internalProxies)) {
584                    // do nothing, internalProxies IPs are not appended to the
585                } else if (matchesOne(currentRemoteIp, trustedProxies)) {
586                    proxiesHeaderValue.addFirst(currentRemoteIp);
587                } else {
588                    idx--; // decrement idx because break statement doesn't do it
589                    break;
590                }
591            }
592            // continue to loop on remoteIpHeaderValue to build the new value of the remoteIpHeader
593            LinkedList<String> newRemoteIpHeaderValue = new LinkedList<String>();
594            for (; idx >= 0; idx--) {
595                String currentRemoteIp = remoteIpHeaderValue[idx];
596                newRemoteIpHeaderValue.addFirst(currentRemoteIp);
597            }
598            if (remoteIp != null) {
599                
600                request.setRemoteAddr(remoteIp);
601                request.setRemoteHost(remoteIp);
602                
603                // use request.coyoteRequest.mimeHeaders.setValue(str).setString(str) because request.addHeader(str, str) is no-op in Tomcat
604                // 6.0
605                if (proxiesHeaderValue.size() == 0) {
606                    request.getCoyoteRequest().getMimeHeaders().removeHeader(proxiesHeader);
607                } else {
608                    String commaDelimitedListOfProxies = listToCommaDelimitedString(proxiesHeaderValue);
609                    request.getCoyoteRequest().getMimeHeaders().setValue(proxiesHeader).setString(commaDelimitedListOfProxies);
610                }
611                if (newRemoteIpHeaderValue.size() == 0) {
612                    request.getCoyoteRequest().getMimeHeaders().removeHeader(remoteIpHeader);
613                } else {
614                    String commaDelimitedRemoteIpHeaderValue = listToCommaDelimitedString(newRemoteIpHeaderValue);
615                    request.getCoyoteRequest().getMimeHeaders().setValue(remoteIpHeader).setString(commaDelimitedRemoteIpHeaderValue);
616                }
617            }
618            
619            if (protocolHeader != null) {
620                String protocolHeaderValue = request.getHeader(protocolHeader);
621                if (protocolHeaderValue == null) {
622                    // don't modify the secure,scheme and serverPort attributes
623                    // of the request
624                } else if (protocolHeaderHttpsValue.equalsIgnoreCase(protocolHeaderValue)) {
625                    request.setSecure(true);
626                    // use request.coyoteRequest.scheme instead of request.setScheme() because request.setScheme() is no-op in Tomcat 6.0
627                    request.getCoyoteRequest().scheme().setString("https");
628                    
629                    request.setServerPort(httpsServerPort);
630                } else {
631                    request.setSecure(false);
632                    // use request.coyoteRequest.scheme instead of request.setScheme() because request.setScheme() is no-op in Tomcat 6.0
633                    request.getCoyoteRequest().scheme().setString("http");
634                    
635                    request.setServerPort(httpServerPort);
636                }
637            }
638            
639            if (log.isDebugEnabled()) {
640                log.debug("Incoming request " + request.getRequestURI() + " with originalRemoteAddr '" + originalRemoteAddr
641                          + "', originalRemoteHost='" + originalRemoteHost + "', originalSecure='" + originalSecure + "', originalScheme='"
642                          + originalScheme + "' will be seen as newRemoteAddr='" + request.getRemoteAddr() + "', newRemoteHost='"
643                          + request.getRemoteHost() + "', newScheme='" + request.getScheme() + "', newSecure='" + request.isSecure() + "'");
644            }
645        } else {
646            if (log.isDebugEnabled()) {
647                log.debug("Skip RemoteIpValve for request " + request.getRequestURI() + " with originalRemoteAddr '"
648                        + request.getRemoteAddr() + "'");
649            }
650        }
651        try {
652            getNext().invoke(request, response);
653        } finally {
654            request.setRemoteAddr(originalRemoteAddr);
655            request.setRemoteHost(originalRemoteHost);
656            
657            request.setSecure(originalSecure);
658            
659            // use request.coyoteRequest.scheme instead of request.setScheme() because request.setScheme() is no-op in Tomcat 6.0
660            request.getCoyoteRequest().scheme().setString(originalScheme);
661            
662            request.setServerPort(originalServerPort);
663        }
664    }
