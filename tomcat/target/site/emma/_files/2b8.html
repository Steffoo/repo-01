588     * need to override this method.
589     *

591     *                  contains the request the client made of
592     *                  the servlet

595     *                  contains the response the servlet returns
596     *                  to the client                                
597     *
598     * @exception IOException   if an input or output error occurs
599     *                              while the servlet is handling the
600     *                              HTTP request
601     *
602     * @exception ServletException  if the HTTP request
603     *                                  cannot be handled
604     * 
605     * @see javax.servlet.Servlet#service
606     */
607    protected void service(HttpServletRequest req, HttpServletResponse resp)
608        throws ServletException, IOException {
609 
610        String method = req.getMethod();
611 
612        if (method.equals(METHOD_GET)) {
613            long lastModified = getLastModified(req);
614            if (lastModified == -1) {
615                // servlet doesn't support if-modified-since, no reason
616                // to go through further expensive logic
617                doGet(req, resp);
618            } else {
619                long ifModifiedSince;
620                try {
621                    ifModifiedSince = req.getDateHeader(HEADER_IFMODSINCE);
622                } catch (IllegalArgumentException iae) {
623                    // Invalid date header - proceed as if none was set
624                    ifModifiedSince = -1;
625                }
626                if (ifModifiedSince < (lastModified / 1000 * 1000)) {
627                    // If the servlet mod time is later, call doGet()
628                    // Round down to the nearest second for a proper compare
629                    // A ifModifiedSince of -1 will always be less
630                    maybeSetLastModified(resp, lastModified);
631                    doGet(req, resp);
632                } else {
633                    resp.setStatus(HttpServletResponse.SC_NOT_MODIFIED);
634                }
635            }
636 
637        } else if (method.equals(METHOD_HEAD)) {
638            long lastModified = getLastModified(req);
639            maybeSetLastModified(resp, lastModified);
640            doHead(req, resp);
641 
642        } else if (method.equals(METHOD_POST)) {
643            doPost(req, resp);
644            
645        } else if (method.equals(METHOD_PUT)) {
646            doPut(req, resp);        
647            
648        } else if (method.equals(METHOD_DELETE)) {
649            doDelete(req, resp);
650            
651        } else if (method.equals(METHOD_OPTIONS)) {
652            doOptions(req,resp);
653            
654        } else if (method.equals(METHOD_TRACE)) {
655            doTrace(req,resp);
656            
657        } else {
658            //
659            // Note that this means NO servlet supports whatever
660            // method was requested, anywhere on this server.
661            //
662 
663            String errMsg = lStrings.getString("http.method_not_implemented");
664            Object[] errArgs = new Object[1];
665            errArgs[0] = method;
666            errMsg = MessageFormat.format(errMsg, errArgs);
667            
668            resp.sendError(HttpServletResponse.SC_NOT_IMPLEMENTED, errMsg);
669        }
670    }
