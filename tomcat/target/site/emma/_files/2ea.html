506 
507    static class Http11ConnectionHandler implements Handler {
508 
509        protected Http11AprProtocol proto;
510        protected AtomicLong registerCount = new AtomicLong(0);
511        protected RequestGroupInfo global = new RequestGroupInfo();
512 
513        protected final Map<Long, Http11AprProcessor> connections =
514                new ConcurrentHashMap<Long, Http11AprProcessor>();
515        protected ConcurrentLinkedQueue<Http11AprProcessor> recycledProcessors =
516            new ConcurrentLinkedQueue<Http11AprProcessor>() {
517            private static final long serialVersionUID = 1L;
518            protected AtomicInteger size = new AtomicInteger(0);
519            @Override
520            public boolean offer(Http11AprProcessor processor) {
521                boolean offer = (proto.processorCache == -1) ? true : (size.get() < proto.processorCache);
522                //avoid over growing our cache or add after we have stopped
523                boolean result = false;
524                if ( offer ) {
525                    result = super.offer(processor);
526                    if ( result ) {
527                        size.incrementAndGet();
528                    }
529                }
530                if (!result) unregister(processor);
531                return result;
532            }
533 
534            @Override
535            public Http11AprProcessor poll() {
536                Http11AprProcessor result = super.poll();
537                if ( result != null ) {
538                    size.decrementAndGet();
539                }
540                return result;
541            }
542 
543            @Override
544            public void clear() {
545                Http11AprProcessor next = poll();
546                while ( next != null ) {
547                    unregister(next);
548                    next = poll();
549                }
550                super.clear();
551                size.set(0);
552            }
553        };
554

556        Http11ConnectionHandler(Http11AprProtocol proto) {
557            this.proto = proto;
558        }
559 
560        public SocketState event(long socket, SocketStatus status) {
561            Http11AprProcessor result = connections.get(Long.valueOf(socket));
562 
563            SocketState state = SocketState.CLOSED;
564            if (result != null) {
565                // Call the appropriate event
566                try {
567                    state = result.event(status);
568                } catch (java.net.SocketException e) {
569                    // SocketExceptions are normal
570                    Http11AprProtocol.log.debug
571                        (sm.getString
572                            ("http11protocol.proto.socketexception.debug"), e);
573                } catch (java.io.IOException e) {
574                    // IOExceptions are normal
575                    Http11AprProtocol.log.debug
576                        (sm.getString
577                            ("http11protocol.proto.ioexception.debug"), e);
578                }
579                // Future developers: if you discover any other
580                // rare-but-nonfatal exceptions, catch them here, and log as
581                // above.
582                catch (Throwable e) {
583                    // any other exception or error is odd. Here we log it
584                    // with "ERROR" level, so it will show up even on
585                    // less-than-verbose logs.
586                    Http11AprProtocol.log.error
587                        (sm.getString("http11protocol.proto.error"), e);
588                } finally {
589                    if (state != SocketState.LONG) {
590                        connections.remove(Long.valueOf(socket));
591                        recycledProcessors.offer(result);
592                        if (state == SocketState.OPEN) {
593                            proto.endpoint.getPoller().add(socket);
594                        }
595                    } else {
596                        proto.endpoint.getCometPoller().add(socket);
597                    }
598                }
599            }
600            return state;
601        }
602 
603        public SocketState process(long socket) {
604            Http11AprProcessor processor = null;
605            try {
606                processor = connections.remove(Long.valueOf(socket));
607 
608                if (processor == null) {
609                    processor = recycledProcessors.poll();
610                }
611 
612                if (processor == null) {
613                    processor = createProcessor();
614                }
615 
616                processor.action(ActionCode.ACTION_START, null);
617 
618                SocketState state = processor.process(socket);
619                if (state == SocketState.LONG) {
620                    // Associate the connection with the processor. The next request
621                    // processed by this thread will use either a new or a recycled
622                    // processor.
623                    connections.put(Long.valueOf(socket), processor);
624                    proto.endpoint.getCometPoller().add(socket);
625                } else if (state == SocketState.OPEN) {
626                    // In keep-alive but between requests. OK to recycle
627                    // processor. Continue to poll for the next request.
628                    recycledProcessors.offer(processor);
629                    proto.endpoint.getPoller().add(socket);
630                } else if (state == SocketState.SENDFILE) {
631                    // Sendfile in progress. If it fails, the socket will be
632                    // closed. If it works, the socket either be added to the
633                    // poller to await more data or processed if there are any
634                    // pipe-lined requests remaining.
635                    connections.put(socket, processor);
636                } else {
637                    recycledProcessors.offer(processor);
638                }
639                return state;
640 
641            } catch (java.net.SocketException e) {
642                // SocketExceptions are normal
643                Http11AprProtocol.log.debug
644                    (sm.getString
645                     ("http11protocol.proto.socketexception.debug"), e);
646            } catch (java.io.IOException e) {
647                // IOExceptions are normal
648                Http11AprProtocol.log.debug
649                    (sm.getString
650                     ("http11protocol.proto.ioexception.debug"), e);
651            }
652            // Future developers: if you discover any other
653            // rare-but-nonfatal exceptions, catch them here, and log as
654            // above.
655            catch (Throwable e) {
656                // any other exception or error is odd. Here we log it
657                // with "ERROR" level, so it will show up even on
658                // less-than-verbose logs.
659                Http11AprProtocol.log.error
660                    (sm.getString("http11protocol.proto.error"), e);
661            }
662            recycledProcessors.offer(processor);
663            return SocketState.CLOSED;
664        }
