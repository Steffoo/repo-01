792        throws IOException {
793        RequestInfo rp = request.getRequestProcessor();
794        rp.setStage(org.apache.coyote.Constants.STAGE_PARSE);
795

797        this.socket = socket;
798        inputBuffer.setSocket(socket);
799        outputBuffer.setSocket(socket);
800        inputBuffer.setSelectorPool(endpoint.getSelectorPool());
801        outputBuffer.setSelectorPool(endpoint.getSelectorPool());
802 
803        // Error flag
804        error = false;
805        keepAlive = true;
806        comet = false;
807        sendfileInProgress = false;
808 
809        int keepAliveLeft = maxKeepAliveRequests;
810        long soTimeout = endpoint.getSoTimeout();
811 
812        boolean keptAlive = false;
813        boolean openSocket = false;
814        boolean recycle = true;
815        while (!error && keepAlive && !comet) {
816 
817            // Parsing the request header
818            try {
819                if( !disableUploadTimeout && keptAlive && soTimeout > 0 ) {
820                    socket.getIOChannel().socket().setSoTimeout((int)soTimeout);
821                }
822                if (!inputBuffer.parseRequestLine(keptAlive)) {
823                    //no data available yet, since we might have read part
824                    //of the request line, we can't recycle the processor
825                    openSocket = true;
826                    recycle = false;
827                    break;
828                }
829                keptAlive = true;
830                // Set this every time in case limit has been changed via JMX
831                request.getMimeHeaders().setLimit(endpoint.getMaxHeaderCount());
832                request.getCookies().setLimit(getMaxCookieCount());
833                if ( !inputBuffer.parseHeaders() ) {
834                    //we've read part of the request, don't recycle it
835                    //instead associate it with the socket
836                    openSocket = true;
837                    recycle = false;
838                    break;
839                }
840                request.setStartTime(System.currentTimeMillis());
841                if (!disableUploadTimeout) { //only for body, not for request headers
842                    socket.getIOChannel().socket().setSoTimeout(timeout);
843                }
844            } catch (IOException e) {
845                if (log.isDebugEnabled()) {
846                    log.debug(sm.getString("http11processor.header.parse"), e);
847                }
848                error = true;
849                break;
850            } catch (Throwable t) {
851                if (log.isDebugEnabled()) {
852                    log.debug(sm.getString("http11processor.header.parse"), t);
853                }

855                response.setStatus(400);
856                adapter.log(request, response, 0);
857                error = true;
858            }
859 
860            if (!error) {
861                // Setting up filters, and parse some request headers
862                rp.setStage(org.apache.coyote.Constants.STAGE_PREPARE);
863                try {
864                    prepareRequest();
865                } catch (Throwable t) {
866                    if (log.isDebugEnabled()) {
867                        log.debug(sm.getString("http11processor.request.prepare"), t);
868                    }
869                    // 400 - Internal Server Error
870                    response.setStatus(400);
871                    adapter.log(request, response, 0);
872                    error = true;
873                }
874            }
875 
876            if (maxKeepAliveRequests > 0 && --keepAliveLeft == 0)
877                keepAlive = false;
878 
879            // Process the request in the adapter
880            if (!error) {
881                try {
882                    rp.setStage(org.apache.coyote.Constants.STAGE_SERVICE);
883                    adapter.service(request, response);

885                    // error occurred.  Throwing a ServletException should both
886                    // set the status to 500 and set the errorException.
887                    // If we fail here, then the response is likely already
888                    // committed, so we can't try and set headers.
889                    if(keepAlive && !error) { // Avoid checking twice.
890                        error = response.getErrorException() != null ||

892                    }

894                    SelectionKey key = socket.getIOChannel().keyFor(socket.getPoller().getSelector());

896                        NioEndpoint.KeyAttachment attach = (NioEndpoint.KeyAttachment) key.attachment();

913                }
914            }

937            }
938 
939            rp.setStage(org.apache.coyote.Constants.STAGE_KEEPALIVE);
940 
941            if (breakKeepAliveLoop(socket)) {
942                break;
943            }
944        }
945 
946        rp.setStage(org.apache.coyote.Constants.STAGE_ENDED);
947 
948        if (comet) {
949            if (error) {
950                recycle();
951                return SocketState.CLOSED;
952            } else {
953                return SocketState.LONG;
954            }
955        } else if (sendfileInProgress) {
956            return SocketState.SENDFILE;
957        } else {
958            if (recycle) {
959                recycle();
960            }
961            return (openSocket) ? (recycle?SocketState.OPEN:SocketState.LONG) : SocketState.CLOSED;
962        }
