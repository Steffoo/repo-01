610    static class Http11ConnectionHandler implements Handler {
611 
612        protected Http11NioProtocol proto;
613        protected static int count = 0;
614        protected RequestGroupInfo global = new RequestGroupInfo();
615 
616        protected final Map<NioChannel, Http11NioProcessor> connections =
617                new ConcurrentHashMap<NioChannel, Http11NioProcessor>();
618        protected ConcurrentLinkedQueue<Http11NioProcessor> recycledProcessors =
619                new ConcurrentLinkedQueue<Http11NioProcessor>() {
620            private static final long serialVersionUID = 1L;
621            protected AtomicInteger size = new AtomicInteger(0);
622            @Override
623            public boolean offer(Http11NioProcessor processor) {
624                boolean offer = proto.processorCache==-1?true:size.get() < proto.processorCache;
625                //avoid over growing our cache or add after we have stopped
626                boolean result = false;
627                if ( offer ) {
628                    result = super.offer(processor);
629                    if ( result ) {
630                        size.incrementAndGet();
631                    }
632                }
633                if (!result) deregister(processor);
634                return result;
635            }
636 
637            @Override
638            public Http11NioProcessor poll() {
639                Http11NioProcessor result = super.poll();
640                if ( result != null ) {
641                    size.decrementAndGet();
642                }
643                return result;
644            }
645 
646            @Override
647            public void clear() {
648                Http11NioProcessor next = poll();
649                while ( next != null ) {
650                    deregister(next);
651                    next = poll();
652                }
653                super.clear();
654                size.set(0);
655            }
656        };
657 
658        Http11ConnectionHandler(Http11NioProtocol proto) {
659            this.proto = proto;
660        }
661 
662        public void releaseCaches() {
663            recycledProcessors.clear();
664        }
665 
666        /**
667         * Use this only if the processor is not available, otherwise use
668         * {@link #release(NioChannel, Http11NioProcessor).
669         */
670        public void release(NioChannel socket) {
671            Http11NioProcessor result = connections.remove(socket);
672            if ( result != null ) {
673                result.recycle();
674                recycledProcessors.offer(result);
675            }
676        }

679        public void release(NioChannel socket, Http11NioProcessor processor) {
680            connections.remove(socket);
681            processor.recycle();
682            recycledProcessors.offer(processor);
683        }

686        public SocketState event(NioChannel socket, SocketStatus status) {
687            Http11NioProcessor result = connections.get(socket);
688 
689            SocketState state = SocketState.CLOSED;
690            if (result != null) {
691                if (log.isDebugEnabled()) log.debug("Http11NioProcessor.error="+result.error);
692                // Call the appropriate event
693                try {
694                    state = result.event(status);
695                } catch (java.net.SocketException e) {
696                    // SocketExceptions are normal
697                    Http11NioProtocol.log.debug
698                        (sm.getString
699                            ("http11protocol.proto.socketexception.debug"), e);
700                } catch (java.io.IOException e) {
701                    // IOExceptions are normal
702                    Http11NioProtocol.log.debug
703                        (sm.getString
704                            ("http11protocol.proto.ioexception.debug"), e);
705                }
706                // Future developers: if you discover any other
707                // rare-but-nonfatal exceptions, catch them here, and log as
708                // above.
709                catch (Throwable e) {
710                    // any other exception or error is odd. Here we log it
711                    // with "ERROR" level, so it will show up even on
712                    // less-than-verbose logs.
713                    Http11NioProtocol.log.error
714                        (sm.getString("http11protocol.proto.error"), e);
715                } finally {
716                    if (state != SocketState.LONG) {
717                        release(socket, result);
718                        if (state == SocketState.OPEN) {
719                            socket.getPoller().add(socket);
720                        }
721                    } else {
722                        if (log.isDebugEnabled()) log.debug("Keeping processor["+result);
723                        //add correct poller events here based on Comet stuff
724                        NioEndpoint.KeyAttachment att = (NioEndpoint.KeyAttachment)socket.getAttachment(false);
725                        socket.getPoller().add(socket,att.getCometOps());
726                    }

732        public SocketState process(NioChannel socket) {
733            Http11NioProcessor processor = null;
734            try {
735                processor = connections.remove(socket);
736 
737                if (processor == null) {
738                    processor = recycledProcessors.poll();
739                }
740                if (processor == null) {
741                    processor = createProcessor();
742                }
743 
744                processor.action(ActionCode.ACTION_START, null);
745 
746                if (proto.ep.isSSLEnabled() && (proto.sslImplementation != null)) {
747                    if (socket instanceof SecureNioChannel) {
748                        SecureNioChannel ch = (SecureNioChannel)socket;
749                        processor.setSslSupport(proto.sslImplementation.getSSLSupport(ch.getSslEngine().getSession()));
750                    }else processor.setSslSupport(null);
751                } else {
752                    processor.setSslSupport(null);
753                }
754 
755 
756                SocketState state = processor.process(socket);
757                if (state == SocketState.LONG) {
758                    // In the middle of processing a request/response. Keep the
759                    // socket associated with the processor.
760                    connections.put(socket, processor);
761                    socket.getPoller().add(socket);
762                } else if (state == SocketState.OPEN) {
763                    // In keep-alive but between requests. OK to recycle
764                    // processor. Continue to poll for the next request.
765                    release(socket, processor);
766                    socket.getPoller().add(socket);
767                } else if (state == SocketState.SENDFILE) {
768                    // Sendfile in progress. If it fails, the socket will be
769                    // closed. If it works, the socket either be added to the
770                    // poller to await more data or processed if there are any
771                    // pipe-lined requests remaining.
772                    connections.put(socket, processor);
773                } else {
774                    // Connection closed. OK to recycle the processor.
775                    release(socket, processor);
776                }
777                return state;
778 
779            } catch (java.net.SocketException e) {
780                // SocketExceptions are normal
781                Http11NioProtocol.log.debug
782                    (sm.getString
783                     ("http11protocol.proto.socketexception.debug"), e);
784            } catch (java.io.IOException e) {
785                // IOExceptions are normal
786                Http11NioProtocol.log.debug
787                    (sm.getString
788                     ("http11protocol.proto.ioexception.debug"), e);
789            }
790            // Future developers: if you discover any other
791            // rare-but-nonfatal exceptions, catch them here, and log as
792            // above.
793            catch (Throwable e) {
794                // any other exception or error is odd. Here we log it
795                // with "ERROR" level, so it will show up even on
796                // less-than-verbose logs.
797                Http11NioProtocol.log.error
798                    (sm.getString("http11protocol.proto.error"), e);
799            }
800            release(socket, processor);
801            return SocketState.CLOSED;
802        }
