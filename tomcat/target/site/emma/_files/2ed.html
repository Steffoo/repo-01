597                headerValue = headers.addValue(buf, start, pos - start);
598            } else if (!HttpParser.isToken(buf[pos])) {
599                // If a non-token header is detected, skip the line and
600                // ignore the header
601                skipLine(start);
602                return true;
603            }
604            chr = buf[pos];
605            if ((chr >= Constants.A) && (chr <= Constants.Z)) {
606                buf[pos] = (byte) (chr - Constants.LC_OFFSET);
607            }
608 
609            pos++;
610 
611        }
612 
613        // Mark the current buffer position
614        start = pos;
615        int realPos = pos;
616 
617        //
618        // Reading the header value (which can be spanned over multiple lines)
619        //
620 
621        boolean eol = false;
622        boolean validLine = true;
623 
624        while (validLine) {
625 
626            boolean space = true;
627 
628            // Skipping spaces
629            while (space) {
630 
631                // Read new bytes if needed
632                if (pos >= lastValid) {
633                    if (!fill())
634                        throw new EOFException(sm.getString("iib.eof.error"));
635                }
636 
637                if ((buf[pos] == Constants.SP) || (buf[pos] == Constants.HT)) {
638                    pos++;
639                } else {
640                    space = false;
641                }
642 
643            }
644 
645            int lastSignificantChar = realPos;
646 
647            // Reading bytes until the end of the line
648            while (!eol) {
649 
650                // Read new bytes if needed
651                if (pos >= lastValid) {
652                    if (!fill())
653                        throw new EOFException(sm.getString("iib.eof.error"));
654                }
655 
656                if (buf[pos] == Constants.CR) {
657                    // Skip
658                } else if (buf[pos] == Constants.LF) {
659                    eol = true;
660                } else if (buf[pos] == Constants.SP) {
661                    buf[realPos] = buf[pos];
662                    realPos++;
663                } else {
664                    buf[realPos] = buf[pos];
665                    realPos++;
666                    lastSignificantChar = realPos;
667                }
668 
669                pos++;
670 
671            }
672 
673            realPos = lastSignificantChar;
674 
675            // Checking the first character of the new line. If the character
676            // is a LWS, then it's a multiline header
677 
678            // Read new bytes if needed
679            if (pos >= lastValid) {
680                if (!fill())
681                    throw new EOFException(sm.getString("iib.eof.error"));
682            }
683 
684            chr = buf[pos];
685            if ((chr != Constants.SP) && (chr != Constants.HT)) {
686                validLine = false;
687            } else {
688                eol = false;
689                // Copying one extra space in the buffer (since there must
690                // be at least one space inserted between the lines)
691                buf[realPos] = chr;
692                realPos++;
693            }
694 
695        }
696 
697        // Set the header value
698        headerValue.setBytes(buf, start, realPos - start);
699 
700        return true;
701 
702    }

705    private void skipLine(int start) throws IOException {
706        boolean eol = false;
707        int lastRealByte = start;
708        if (pos - 1 > start) {
709            lastRealByte = pos - 1;
710        }
711 
712        while (!eol) {
713 
714            // Read new bytes if needed
715            if (pos >= lastValid) {
716                if (!fill())
717                    throw new EOFException(sm.getString("iib.eof.error"));
718            }
719 
720            if (buf[pos] == Constants.CR) {
721                // Skip
722            } else if (buf[pos] == Constants.LF) {
723                eol = true;
724            } else {
725                lastRealByte = pos;
726            }
727            pos++;
728        }
729 
730        if (log.isDebugEnabled()) {
731            log.debug(sm.getString("iib.invalidheader", new String(buf, start,
732                    lastRealByte - start + 1, "ISO-8859-1")));
733        }

771    /**
772     * Fill the internal buffer using data from the undelying input stream.
773     *
774     * @return false if at end of stream
775     */
776    protected boolean fill()
777        throws IOException {
778 
779        int nRead = 0;
780 
781        if (parsingHeader) {
782 
783            if (lastValid == buf.length) {
784                throw new IllegalArgumentException
785                    (sm.getString("iib.requestheadertoolarge.error"));
786            }
787 
788            bbuf.clear();
789            nRead = Socket.recvbb(socket, 0, buf.length - lastValid);
790            if (nRead > 0) {
791                bbuf.limit(nRead);
792                bbuf.get(buf, pos, nRead);
793                lastValid = pos + nRead;
794            } else {
795                if ((-nRead) == Status.EAGAIN) {
796                    return false;
797                } else {
798                    throw new IOException(sm.getString("iib.failedread"));
799                }
800            }
801 
802        } else {
803 
804            if (buf.length - end < 4500) {
805                // In this case, the request header was really large, so we allocate a
806                // brand new one; the old one will get GCed when subsequent requests
807                // clear all references
808                buf = new byte[buf.length];
809                end = 0;
810            }
811            pos = end;
812            lastValid = pos;
813            bbuf.clear();
814            nRead = Socket.recvbb(socket, 0, buf.length - lastValid);
815            if (nRead > 0) {
816                bbuf.limit(nRead);
817                bbuf.get(buf, pos, nRead);
818                lastValid = pos + nRead;
819            } else {
820                if ((-nRead) == Status.ETIMEDOUT || (-nRead) == Status.TIMEUP) {
821                    throw new SocketTimeoutException(sm.getString("iib.failedread"));
822                } else if (nRead == 0) {
823                    // APR_STATUS_IS_EOF, since native 1.1.22
824                    return false;
825                } else {
826                    throw new IOException(sm.getString("iib.failedread"));
827                }
828            }
829 
830        }
831 
832        return (nRead > 0);
833 
834    }
