539            boolean space = false;

542                if (pos >= lastValid) {

544                        return false;

572        }

586                }

592            end = 0;
593        }
594        if (parsingRequestLinePhase == 6) {

603                        return false;
604                }

611                    parsingRequestLineEol = true;
612                } else if (!HttpParser.isHttpProtocol(buf[pos])) {

616            }

673        }

728    public HeaderParseStatus parseHeader()

732        // Check for blank line
733        //
734 
735        byte chr = 0;
736        while (headerParsePos == HeaderParsePosition.HEADER_START) {
737

739            if (pos >= lastValid) {

744            }

747 
748            if (chr == Constants.CR) {
749                // Skip
750            } else if (chr == Constants.LF) {

756 
757            pos++;
758 
759        }
760

762            // Mark the current buffer position
763            headerData.start = pos;
764            headerParsePos = HeaderParsePosition.HEADER_NAME;
765        }
766 
767        //
768        // Reading the header name
769        // Header name is always US-ASCII
770        //
771 
772        while (headerParsePos == HeaderParsePosition.HEADER_NAME) {
773 
774            // Read new bytes if needed
775            if (pos >= lastValid) {
776                if (!fill(true,false)) { //parse header
777                    return HeaderParseStatus.NEED_MORE_DATA;
778                }
779            }
780 
781            chr = buf[pos];
782            if (chr == Constants.COLON) {
783                headerParsePos = HeaderParsePosition.HEADER_VALUE_START;
784                headerData.headerValue = headers.addValue(buf, headerData.start, pos - headerData.start);
785                pos++;
786                // Mark the current buffer position
787                headerData.start = pos;
788                headerData.realPos = pos;
789                headerData.lastSignificantChar = pos;
790                break;
791            } else if (!HttpParser.isToken(chr)) {
792                // If a non-token header is detected, skip the line and
793                // ignore the header
794                headerData.lastSignificantChar = pos;
795                return skipLine();
796            }
797 
798            // chr is next byte of header name. Convert to lowercase.
799            if ((chr >= Constants.A) && (chr <= Constants.Z)) {
800                buf[pos] = (byte) (chr - Constants.LC_OFFSET);
801            }
802            pos++;
803        }
804 
805        // Skip the line and ignore the header
806        if (headerParsePos == HeaderParsePosition.HEADER_SKIPLINE) {
807            return skipLine();
808        }
809 
810        //
811        // Reading the header value (which can be spanned over multiple lines)
812        //
813 
814        while (headerParsePos == HeaderParsePosition.HEADER_VALUE_START ||
815               headerParsePos == HeaderParsePosition.HEADER_VALUE ||
816               headerParsePos == HeaderParsePosition.HEADER_MULTI_LINE) {
817 
818            if ( headerParsePos == HeaderParsePosition.HEADER_VALUE_START ) {
819                // Skipping spaces
820                while (true) {
821                    // Read new bytes if needed
822                    if (pos >= lastValid) {
823                        if (!fill(true,false)) {//parse header
824                            //HEADER_VALUE_START
825                            return HeaderParseStatus.NEED_MORE_DATA;
826                        }
827                    }
828 
829                    chr = buf[pos];
830                    if (chr == Constants.SP || chr == Constants.HT) {
831                        pos++;
832                    } else {
833                        headerParsePos = HeaderParsePosition.HEADER_VALUE;
834                        break;
835                    }
836                }
837            }
838            if ( headerParsePos == HeaderParsePosition.HEADER_VALUE ) {
839 
840                // Reading bytes until the end of the line
841                boolean eol = false;
842                while (!eol) {
843 
844                    // Read new bytes if needed
845                    if (pos >= lastValid) {
846                        if (!fill(true,false)) {//parse header
847                            //HEADER_VALUE
848                            return HeaderParseStatus.NEED_MORE_DATA;
849                        }
850                    }
851 
852                    chr = buf[pos];
853                    if (chr == Constants.CR) {
854                        // Skip
855                    } else if (chr == Constants.LF) {
856                        eol = true;
857                    } else if (chr == Constants.SP || chr == Constants.HT) {
858                        buf[headerData.realPos] = chr;
859                        headerData.realPos++;
860                    } else {
861                        buf[headerData.realPos] = chr;
862                        headerData.realPos++;
863                        headerData.lastSignificantChar = headerData.realPos;
864                    }
865 
866                    pos++;
867                }
868

875            }
876            // Read new bytes if needed
877            if (pos >= lastValid) {
878                if (!fill(true,false)) {//parse header
879 
880                    //HEADER_MULTI_LINE
881                    return HeaderParseStatus.NEED_MORE_DATA;
882                }
883            }
884 
885            chr = buf[pos];
886            if ( headerParsePos == HeaderParsePosition.HEADER_MULTI_LINE ) {

888                    headerParsePos = HeaderParsePosition.HEADER_START;

892                    // be at least one space inserted between the lines)
893                    buf[headerData.realPos] = chr;
894                    headerData.realPos++;
895                    headerParsePos = HeaderParsePosition.HEADER_VALUE_START;
896                }
897            }
898        }
899        // Set the header value
900        headerData.headerValue.setBytes(buf, headerData.start,
901                headerData.lastSignificantChar - headerData.start);
902        headerData.recycle();
903        return HeaderParseStatus.HAVE_MORE_HEADERS;
