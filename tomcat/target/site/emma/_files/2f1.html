746    /**
747     * Process pipelined HTTP requests on the specified socket.
748     *
749     * @param theSocket Socket from which the HTTP requests will be read
750     *               and the HTTP responses will be written.
751     *
752     * @throws IOException error during an I/O operation
753     */
754    public void process(Socket theSocket)
755        throws IOException {
756        RequestInfo rp = request.getRequestProcessor();
757        rp.setStage(org.apache.coyote.Constants.STAGE_PARSE);
758 
759        // Set the remote address
760        remoteAddr = null;
761        remoteHost = null;
762        localAddr = null;
763        localName = null;
764        remotePort = -1;
765        localPort = -1;
766 
767        // Setting up the I/O
768        this.socket = theSocket;
769        inputBuffer.setInputStream(socket.getInputStream());
770        outputBuffer.setOutputStream(socket.getOutputStream());
771 
772        // Error flag
773        error = false;
774        keepAlive = true;
775 
776        int keepAliveLeft = maxKeepAliveRequests;
777        int soTimeout = endpoint.getSoTimeout();
778 
779        // When using an executor, these values may return non-positive values
780        int curThreads = endpoint.getCurrentThreadsBusy();
781        int maxThreads = endpoint.getMaxThreads();
782        if (curThreads > 0 && maxThreads > 0) {
783            // Only auto-disable keep-alive if the current thread usage % can be
784            // calculated correctly
785            if ((curThreads*100)/maxThreads > 75) {
786                keepAliveLeft = 1;
787            }
788        }
789 
790        try {
791            socket.setSoTimeout(soTimeout);
792        } catch (Throwable t) {
793            log.debug(sm.getString("http11processor.socket.timeout"), t);
794            error = true;
795        }
796 
797        boolean keptAlive = false;
798 
799        while (started && !error && keepAlive && !endpoint.isPaused()) {
800 
801            // Parsing the request header
802            try {
803                if (keptAlive) {
804                    if (keepAliveTimeout > 0) {
805                        socket.setSoTimeout(keepAliveTimeout);
806                    }
807                    else if (soTimeout > 0) {
808                        socket.setSoTimeout(soTimeout);
809                    }
810                }
811                inputBuffer.parseRequestLine();
812                request.setStartTime(System.currentTimeMillis());
813                keptAlive = true;
814                if (disableUploadTimeout) {
815                    socket.setSoTimeout(soTimeout);
816                } else {
817                    socket.setSoTimeout(timeout);
818                }
819                // Set this every time in case limit has been changed via JMX
820                request.getMimeHeaders().setLimit(endpoint.getMaxHeaderCount());
821                request.getCookies().setLimit(getMaxCookieCount());
822                inputBuffer.parseHeaders();
823            } catch (IOException e) {
824                error = true;
825                break;
826            } catch (Throwable t) {
827                if (log.isDebugEnabled()) {
828                    log.debug(sm.getString("http11processor.header.parse"), t);
829                }
830                // 400 - Bad Request
831                response.setStatus(400);
832                adapter.log(request, response, 0);
833                error = true;
834            }
835 
836            if (!error) {
837                // Setting up filters, and parse some request headers
838                rp.setStage(org.apache.coyote.Constants.STAGE_PREPARE);
839                try {
840                    prepareRequest();
841                } catch (Throwable t) {
842                    if (log.isDebugEnabled()) {
843                        log.debug(sm.getString("http11processor.request.prepare"), t);
844                    }
845                    // 400 - Internal Server Error
846                    response.setStatus(400);
847                    adapter.log(request, response, 0);
848                    error = true;
849                }
850            }
851 
852            if (maxKeepAliveRequests > 0 && --keepAliveLeft == 0)
853                keepAlive = false;
854 
855            // Process the request in the adapter
856            if (!error) {
857                try {
858                    rp.setStage(org.apache.coyote.Constants.STAGE_SERVICE);
859                    adapter.service(request, response);
860                    // Handle when the response was committed before a serious
861                    // error occurred.  Throwing a ServletException should both
862                    // set the status to 500 and set the errorException.
863                    // If we fail here, then the response is likely already
864                    // committed, so we can't try and set headers.
865                    if(keepAlive && !error) { // Avoid checking twice.
866                        error = response.getErrorException() != null ||
867                                statusDropsConnection(response.getStatus());
868                    }
869 
870                } catch (InterruptedIOException e) {
871                    error = true;
872                } catch (Throwable t) {
873                    log.error(sm.getString("http11processor.request.process"), t);
874                    // 500 - Internal Server Error
875                    response.setStatus(500);
876                    adapter.log(request, response, 0);
877                    error = true;
878                }
879            }
880 
881            // Finish the handling of the request
882            try {
883                rp.setStage(org.apache.coyote.Constants.STAGE_ENDINPUT);
884                // If we know we are closing the connection, don't drain input.
885                // This way uploading a 100GB file doesn't tie up the thread
886                // if the servlet has rejected it.
887                if(error)
888                    inputBuffer.setSwallowInput(false);
889                inputBuffer.endRequest();
890            } catch (IOException e) {
891                error = true;
892            } catch (Throwable t) {
893                log.error(sm.getString("http11processor.request.finish"), t);
894                // 500 - Internal Server Error
895                response.setStatus(500);
896                // No access logging since after service method
897                error = true;
898            }
899            try {
900                rp.setStage(org.apache.coyote.Constants.STAGE_ENDOUTPUT);
901                outputBuffer.endRequest();
902            } catch (IOException e) {
903                error = true;
904            } catch (Throwable t) {
905                log.error(sm.getString("http11processor.response.finish"), t);
906                error = true;
907            }
908 
909            // If there was an error, make sure the request is counted as
910            // and error, and update the statistics counter
911            if (error) {
912                response.setStatus(500);
913            }
914            request.updateCounters();
915 
916            rp.setStage(org.apache.coyote.Constants.STAGE_KEEPALIVE);
917 
918            // Don't reset the param - we'll see it as ended. Next request
919            // will reset it
920            // thrA.setParam(null);
921            // Next request
922            inputBuffer.nextRequest();
923            outputBuffer.nextRequest();
924 
925        }
926 
927        rp.setStage(org.apache.coyote.Constants.STAGE_ENDED);
928 
929        // Recycle
930        inputBuffer.recycle();
931        outputBuffer.recycle();
932        this.socket = null;
933        // Recycle ssl info
934        sslSupport = null;
935    }
