543    protected static class Http11ConnectionHandler implements Handler {
544 
545        protected Http11Protocol proto;
546        protected AtomicLong registerCount = new AtomicLong(0);
547        protected RequestGroupInfo global = new RequestGroupInfo();
548 
549        protected ConcurrentLinkedQueue<Http11Processor> recycledProcessors =
550                new ConcurrentLinkedQueue<Http11Processor>() {
551            private static final long serialVersionUID = 1L;
552            protected AtomicInteger size = new AtomicInteger(0);
553            @Override
554            public boolean offer(Http11Processor processor) {
555                boolean offer = (proto.processorCache == -1) ? true : (size.get() < proto.processorCache);
556                //avoid over growing our cache or add after we have stopped
557                boolean result = false;
558                if ( offer ) {
559                    result = super.offer(processor);
560                    if ( result ) {
561                        size.incrementAndGet();
562                    }
563                }
564                if (!result) unregister(processor);
565                return result;
566            }
567 
568            @Override
569            public Http11Processor poll() {
570                Http11Processor result = super.poll();
571                if ( result != null ) {
572                    size.decrementAndGet();
573                }
574                return result;
575            }
576 
577            @Override
578            public void clear() {
579                Http11Processor next = poll();
580                while ( next != null ) {
581                    unregister(next);
582                    next = poll();
583                }
584                super.clear();
585                size.set(0);
586            }
587        };
588 
589        Http11ConnectionHandler(Http11Protocol proto) {
590            this.proto = proto;
591        }

593        public boolean process(Socket socket) {
594            Http11Processor processor = recycledProcessors.poll();
595            try {
596 
597                if (processor == null) {
598                    processor = createProcessor();
599                }
600 
601                processor.action(ActionCode.ACTION_START, null);
602 
603                if (proto.isSSLEnabled() && (proto.sslImplementation != null)) {
604                    processor.setSSLSupport
605                        (proto.sslImplementation.getSSLSupport(socket));
606                } else {
607                    processor.setSSLSupport(null);
608                }
609 
610                processor.process(socket);
611                return false;
612 
613            } catch(java.net.SocketException e) {
614                // SocketExceptions are normal
615                Http11Protocol.log.debug
616                    (sm.getString
617                     ("http11protocol.proto.socketexception.debug"), e);
618            } catch (java.io.IOException e) {
619                // IOExceptions are normal
620                Http11Protocol.log.debug
621                    (sm.getString
622                     ("http11protocol.proto.ioexception.debug"), e);
623            }
624            // Future developers: if you discover any other
625            // rare-but-nonfatal exceptions, catch them here, and log as
626            // above.
627            catch (Throwable e) {
628                // any other exception or error is odd. Here we log it
629                // with "ERROR" level, so it will show up even on
630                // less-than-verbose logs.
631                Http11Protocol.log.error
632                    (sm.getString("http11protocol.proto.error"), e);
633            } finally {
634                processor.action(ActionCode.ACTION_STOP, null);
635                recycledProcessors.offer(processor);
636            }
637            return false;
638        }

661        protected void register(Http11Processor processor) {
662            if (proto.getDomain() != null) {
663                synchronized (this) {
664                    try {
665                        long count = registerCount.incrementAndGet();
666                        RequestInfo rp = processor.getRequest().getRequestProcessor();
667                        rp.setGlobalProcessor(global);
668                        ObjectName rpName = new ObjectName
669                            (proto.getDomain() + ":type=RequestProcessor,worker="
670                                + proto.getName() + ",name=HttpRequest" + count);
671                        if (log.isDebugEnabled()) {
672                            log.debug("Register " + rpName);
673                        }
674                        Registry.getRegistry(null, null).registerComponent(rp, rpName, null);
675                        rp.setRpName(rpName);
676                    } catch (Exception e) {
677                        log.warn("Error registering request");
678                    }
679                }
680            }
681        }

684            if (proto.getDomain() != null) {
685                synchronized (this) {
686                    try {
687                        RequestInfo rp = processor.getRequest().getRequestProcessor();
688                        rp.setGlobalProcessor(null);

691                            log.debug("Unregister " + rpName);
692                        }
693                        Registry.getRegistry(null, null).unregisterComponent(rpName);
694                        rp.setRpName(null);
695                    } catch (Exception e) {
