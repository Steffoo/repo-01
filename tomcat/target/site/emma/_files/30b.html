769    /**
770     * Look for a session in the Store and, if found, restore
771     * it in the Manager's list of active sessions if appropriate.
772     * The session will be removed from the Store after swapping
773     * in, but will not be added to the active session list if it
774     * is invalid or past its expiration.
775     */
776    protected Session swapIn(String id) throws IOException {
777 
778        if (store == null)
779            return null;
780 
781        Object swapInLock = null;
782 
783        /*
784         * The purpose of this sync and these locks is to make sure that a
785         * session is only loaded once. It doesn't matter if the lock is removed
786         * and then another thread enters this method and tries to load the same
787         * session. That thread will re-create a swapIn lock for that session,
788         * quickly find that the session is already in sessions, use it and
789         * carry on.
790         */
791        synchronized (this) {
792            swapInLock = sessionSwapInLocks.get(id);
793            if (swapInLock == null) {
794                swapInLock = new Object();
795                sessionSwapInLocks.put(id, swapInLock);
796            }
797        }
798 
799        Session session = null;
800 
801        synchronized (swapInLock) {
802            // First check to see if another thread has loaded the session into
803            // the manager
804            session = sessions.get(id);
805 
806            if (session == null) {
807                try {
808                    if (SecurityUtil.isPackageProtectionEnabled()){
809                        try {
810                            session = (Session) AccessController.doPrivileged(
811                                    new PrivilegedStoreLoad(id));
812                        } catch (PrivilegedActionException ex) {
813                            Exception e = ex.getException();
814                            log.error(sm.getString(
815                                    "persistentManager.swapInException", id),
816                                    e);
817                            if (e instanceof IOException){
818                                throw (IOException)e;
819                            } else if (e instanceof ClassNotFoundException) {
820                                throw (ClassNotFoundException)e;
821                            }
822                        }
823                    } else {
824                         session = store.load(id);
825                    }
826                } catch (ClassNotFoundException e) {
827                    String msg = sm.getString(
828                            "persistentManager.deserializeError", id);
829                    log.error(msg, e);
830                    throw new IllegalStateException(msg, e);
831                }
832 
833                if (session != null && !session.isValid()) {
834                    log.error(sm.getString(
835                            "persistentManager.swapInInvalid", id));
836                    session.expire();
837                    removeSession(id);
838                    session = null;
839                }
840 
841                if (session != null) {
842                    if(log.isDebugEnabled())
843                        log.debug(sm.getString("persistentManager.swapIn", id));
844 
845                    session.setManager(this);
846                    // make sure the listeners know about it.
847                    ((StandardSession)session).tellNew();
848                    add(session);
849                    ((StandardSession)session).activate();
850                    // endAccess() to ensure timeouts happen correctly.
851                    // access() to keep access count correct or it will end up
852                    // negative
853                    session.access();
854                    session.endAccess();
855                }
856            }
857        }
858 
859        // Make sure the lock is removed
860        synchronized (this) {
861            sessionSwapInLocks.remove(id);
862        }
863 
864        return (session);
865 
866    }
