661    /**
662     * Perform the internal processing required to invalidate this session,
663     * without triggering an exception if the session has already expired.
664     *
665     * @param notify Should we notify listeners about the demise of
666     *  this session?
667     */
668    public void expire(boolean notify) {
669 
670        // Check to see if expire is in progress or has previously been called
671        if (expiring || !isValid)
672            return;
673 
674        synchronized (this) {
675            // Check again, now we are inside the sync so this code only runs once
676            // Double check locking - expiring and isValid need to be volatile
677            if (expiring || !isValid)
678                return;
679 
680            if (manager == null)
681                return;
682 
683            // Mark this session as "being expired"
684            expiring = true;
685        
686            // Notify interested application event listeners
687            // FIXME - Assumes we call listeners in reverse order
688            Context context = (Context) manager.getContainer();
689            
690            // The call to expire() may not have been triggered by the webapp.
691            // Make sure the webapp's class loader is set when calling the
692            // listeners
693            ClassLoader oldTccl = null;
694            if (context.getLoader() != null &&
695                    context.getLoader().getClassLoader() != null) {
696                oldTccl = Thread.currentThread().getContextClassLoader();
697                if (Globals.IS_SECURITY_ENABLED) {
698                    PrivilegedAction<Void> pa = new PrivilegedSetTccl(
699                            context.getLoader().getClassLoader());
700                    AccessController.doPrivileged(pa);
701                } else {
702                    Thread.currentThread().setContextClassLoader(
703                            context.getLoader().getClassLoader());
704                }
705            }
706            try {
707                Object listeners[] = context.getApplicationLifecycleListeners();
708                if (notify && (listeners != null)) {
709                    HttpSessionEvent event =
710                        new HttpSessionEvent(getSession());
711                    for (int i = 0; i < listeners.length; i++) {
712                        int j = (listeners.length - 1) - i;
713                        if (!(listeners[j] instanceof HttpSessionListener))
714                            continue;
715                        HttpSessionListener listener =
716                            (HttpSessionListener) listeners[j];
717                        try {
718                            fireContainerEvent(context,
719                                               "beforeSessionDestroyed",
720                                               listener);
721                            listener.sessionDestroyed(event);
722                            fireContainerEvent(context,
723                                               "afterSessionDestroyed",
724                                               listener);
725                        } catch (Throwable t) {
726                            try {
727                                fireContainerEvent(context,
728                                                   "afterSessionDestroyed",
729                                                   listener);
730                            } catch (Exception e) {
731                                // Ignore
732                            }
733                            manager.getContainer().getLogger().error
734                                (sm.getString("standardSession.sessionEvent"), t);
735                        }
736                    }
737                }
738            } finally {
739                if (oldTccl != null) {
740                    if (Globals.IS_SECURITY_ENABLED) {
741                        PrivilegedAction<Void> pa =
742                            new PrivilegedSetTccl(oldTccl);
743                        AccessController.doPrivileged(pa);
744                    } else {
745                        Thread.currentThread().setContextClassLoader(oldTccl);
746                    }
747                }
748            }
749 
750            if (ACTIVITY_CHECK) {
751                accessCount.set(0);
752            }
753            setValid(false);
754 
755            /*
756             * Compute how long this session has been alive, and update
757             * session manager's related properties accordingly
758             */
759            long timeNow = System.currentTimeMillis();
760            int timeAlive = (int) ((timeNow - creationTime)/1000);
761            synchronized (manager) {
762                if (timeAlive > manager.getSessionMaxAliveTime()) {
763                    manager.setSessionMaxAliveTime(timeAlive);
764                }
765                int numExpired = manager.getExpiredSessions();
766                if (numExpired < Integer.MAX_VALUE) {
767                    numExpired++;
768                    manager.setExpiredSessions(numExpired);
769                }
770 
771                int average = manager.getSessionAverageAliveTime();
772                // Using long, as otherwise (average * numExpired) might overflow 
773                average = (int) (((((long) average) * (numExpired - 1)) + timeAlive)
774                        / numExpired);
775                manager.setSessionAverageAliveTime(average);
776            }
777 
778            if (manager instanceof ManagerBase) {
779                ManagerBase mb = (ManagerBase) manager;
780                SessionTiming timing = new SessionTiming(timeNow, timeAlive);
781                synchronized (mb.sessionExpirationTiming) {
782                    mb.sessionExpirationTiming.add(timing);
783                    mb.sessionExpirationTiming.poll();
784                }
785            }
786 
787            // Remove this session from our manager's active sessions
788            manager.remove(this);
789 
790            // Notify interested session event listeners
791            if (notify) {
792                fireSessionEvent(Session.SESSION_DESTROYED_EVENT, null);
793            }
794 
795            // Call the logout method
796            if (principal instanceof GenericPrincipal) {
797                GenericPrincipal gp = (GenericPrincipal) principal;
798                try {
799                    gp.logout();
800                } catch (Exception e) {
801                    manager.getContainer().getLogger().error(
802                            sm.getString("standardSession.logoutfail"),
803                            e);
804                }
805            }
806 
807            // We have completed expire of this session
808            expiring = false;
809 
810            // Unbind any objects associated with this session
811            String keys[] = keys();
812            for (int i = 0; i < keys.length; i++)
813                removeAttributeInternal(keys[i], notify);
814 
815        }
816 
817    }
