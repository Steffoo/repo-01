558                    throwEOFException(sm.getString("chunkedInputFilter.eosTrailer"));
559                }
560            }
561    
562            chr = buf[pos];
563            if ((chr != Constants.SP) && (chr != Constants.HT)) {
564                validLine = false;
565            } else {
566                eol = false;
567                // Copying one extra space in the buffer (since there must
568                // be at least one space inserted between the lines)
569                trailingHeaders.append(chr);
570            }
571    
572        }
573    
574        String headerName = new String(trailingHeaders.getBytes(), startPos,
575                colonPos - startPos, "ISO_8859_1");
576        if (org.apache.coyote.Constants.ALLOWED_TRAILER_HEADERS.contains(
577                headerName.toLowerCase(Locale.ENGLISH))) {
578            MessageBytes headerValue = headers.addValue(headerName);
579            
580            // Set the header value
581            headerValue.setBytes(trailingHeaders.getBytes(), colonPos,
582                    lastSignificantChar - colonPos);
583        }
584 
585        return true;
586    }
