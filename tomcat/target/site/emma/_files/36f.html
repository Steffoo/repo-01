491    /**
492     * Override the default implementation. If no access log is defined for the
493     * Engine, look for one in the Engine's default host and then the default
494     * host's ROOT context. If still none is found, return the default NoOp
495     * access log.
496     */
497    public void logAccess(Request request, Response response, long time,
498            boolean useDefault) {
499 
500        boolean logged = false;
501        
502        if (getAccessLog() != null) {
503            accessLog.log(request, response, time);
504            logged = true;
505        }
506 
507        if (!logged && useDefault) {
508            AccessLog newDefaultAccessLog = defaultAccessLog.get();
509            if (newDefaultAccessLog == null) {
510                // If we reached this point, this Engine can't have an AccessLog
511                // Look in the defaultHost
512                Host host = (Host) findChild(getDefaultHost());
513                Context context = null;
514                boolean checkHost = (host != null);
515                if (checkHost && host instanceof ContainerBase) {
516                    checkHost = ((ContainerBase) host).started;
517                }
518                if (checkHost) {
519                    newDefaultAccessLog = host.getAccessLog();
520 
521                    if (newDefaultAccessLog != null) {
522                        if (defaultAccessLog.compareAndSet(null,
523                                newDefaultAccessLog)) {
524                            AccessLogListener l = new AccessLogListener(this,
525                                    host, null);
526                            l.install();
527                        }
528                    } else {
529                        // Try the ROOT context of default host
530                        context = (Context) host.findChild("");
531                        boolean checkContext = (context != null);
532                        if (checkContext && context instanceof ContainerBase) {
533                            checkContext = ((ContainerBase) context).started;
534                        }
535                        if (checkContext) {
536                            newDefaultAccessLog = context.getAccessLog();
537                            if (newDefaultAccessLog != null) {
538                                if (defaultAccessLog.compareAndSet(null,
539                                        newDefaultAccessLog)) {
540                                    AccessLogListener l = new AccessLogListener(
541                                            this, null, context);
542                                    l.install();
543                                }
544                            }
545                        }
546                    }
547                }
548 
549                if (newDefaultAccessLog == null) {
550                    newDefaultAccessLog = new NoopAccessLog();
551                    if (defaultAccessLog.compareAndSet(null,
552                            newDefaultAccessLog)) {
553                        AccessLogListener l = new AccessLogListener(this, host,
554                                context);
555                        l.install();
556                    }
557                }
558            }
559 
560            newDefaultAccessLog.log(request, response, time);
561        }
562    }
