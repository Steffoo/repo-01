776            PrivilegedAction dp =
777                new PrivilegedAddChild(child);
778            AccessController.doPrivileged(dp);
779        } else {
780            addChildInternal(child);

783 
784    private void addChildInternal(Container child) {
785 
786        if( log.isDebugEnabled() )
787            log.debug("Add child " + child + " " + this);
788        synchronized(children) {
789            if (children.get(child.getName()) != null)
790                throw new IllegalArgumentException("addChild:  Child name '" +
791                                                   child.getName() +
792                                                   "' is not unique");
793            child.setParent(this);  // May throw IAE
794            children.put(child.getName(), child);
795        }
796 
797        // Start child
798        // Don't do this inside sync block - start can be a slow process and
799        // locking the children object can cause problems elsewhere
800        if (started && startChildren && (child instanceof Lifecycle)) {
801            boolean success = false;
802            try {
803                ((Lifecycle) child).start();
804                success = true;
805            } catch (LifecycleException e) {
806                log.error("ContainerBase.addChild: start: ", e);
807                throw new IllegalStateException
808                    ("ContainerBase.addChild: start: " + e);
809            } finally {
810                if (!success) {
811                    synchronized(children) {
812                        children.remove(child.getName());
813                    }
814                }
815            }
816        }
817 
818        fireContainerEvent(ADD_CHILD_EVENT, child);
819    }
