591    protected void parsePathParameters(org.apache.coyote.Request req,
592            Request request) {
593

595        req.decodedURI().toBytes();
596

598        int semicolon = uriBC.indexOf(';', 0);

601        // encoding that doesn't give the expected result so be explicit

611                    String.valueOf(semicolon)));
612            log.debug(sm.getString("coyoteAdapter.debug", "enc", enc));
613        }

617        while (semicolon > -1) {

619            int start = uriBC.getStart();

621 
622            int pathParamStart = semicolon + 1;
623            int pathParamEnd = ByteChunk.findBytes(uriBC.getBuffer(),
624                    start + pathParamStart, end,

636                                enc));

638                    }
639                }

647                        end - start - pathParamEnd + semicolon);

651                                (end - start) - pathParamStart, enc));

654                        log.warn(sm.getString("coyoteAdapter.parsePathParam",
655                                enc));
656                        warnedEncoding = true;
657                    }
658                }
659                uriBC.setEnd(start + semicolon);
660            }
661

663                log.debug(sm.getString("coyoteAdapter.debug", "pathParamStart",
664                        String.valueOf(pathParamStart)));
665                log.debug(sm.getString("coyoteAdapter.debug", "pathParamEnd",
666                        String.valueOf(pathParamEnd)));
667                log.debug(sm.getString("coyoteAdapter.debug", "pv", pv));
668            }

671                int equals = pv.indexOf('=');

674                    String value = pv.substring(equals + 1);
675                    request.addPathParameter(name, value);
676                    if (log.isDebugEnabled()) {
677                        log.debug(sm.getString("coyoteAdapter.debug", "equals",
678                                String.valueOf(equals)));
679                        log.debug(sm.getString("coyoteAdapter.debug", "name",
680                                name));
681                        log.debug(sm.getString("coyoteAdapter.debug", "value",
682                                value));
683                    }
684                }
685            }
686 
687            semicolon = uriBC.indexOf(';', semicolon);
688        }

843    /**
844     * Normalize URI.
845     * <p>
846     * This method normalizes "\", "//", "/./" and "/../". This method will
847     * return false when trying to go above the root, or if the URI contains
848     * a null byte.
849     *
850     * @param uriMB URI to be normalized
851     */
852    public static boolean normalize(MessageBytes uriMB) {
853 
854        ByteChunk uriBC = uriMB.getByteChunk();
855        final byte[] b = uriBC.getBytes();
856        final int start = uriBC.getStart();
857        int end = uriBC.getEnd();
858 
859        // An empty URL is not acceptable
860        if (start == end)
861            return false;
862 
863        // URL * is acceptable
864        if ((end - start == 1) && b[start] == (byte) '*')
865          return true;
866 
867        int pos = 0;
868        int index = 0;
869 
870        // Replace '\' with '/'
871        // Check for null byte
872        for (pos = start; pos < end; pos++) {
873            if (b[pos] == (byte) '\\') {
874                if (ALLOW_BACKSLASH) {
875                    b[pos] = (byte) '/';
876                } else {
877                    return false;
878                }
879            }
880            if (b[pos] == (byte) 0) {
881                return false;
882            }
883        }
884 
885        // The URL must start with '/'
886        if (b[start] != (byte) '/') {
887            return false;
888        }
889 
890        // Replace "//" with "/"
891        for (pos = start; pos < (end - 1); pos++) {
892            if (b[pos] == (byte) '/') {
893                while ((pos + 1 < end) && (b[pos + 1] == (byte) '/')) {
894                    copyBytes(b, pos, pos + 1, end - pos - 1);
895                    end--;
896                }
897            }
898        }
899 
900        // If the URI ends with "/." or "/..", then we append an extra "/"
901        // Note: It is possible to extend the URI by 1 without any side effect
902        // as the next character is a non-significant WS.
903        if (((end - start) >= 2) && (b[end - 1] == (byte) '.')) {
904            if ((b[end - 2] == (byte) '/')
905                || ((b[end - 2] == (byte) '.')
906                    && (b[end - 3] == (byte) '/'))) {
907                b[end] = (byte) '/';
908                end++;
909            }
910        }
911 
912        uriBC.setEnd(end);
913 
914        index = 0;
915 
916        // Resolve occurrences of "/./" in the normalized path
917        while (true) {
918            index = uriBC.indexOf("/./", 0, 3, index);
919            if (index < 0)
920                break;
921            copyBytes(b, start + index, start + index + 2,
922                      end - start - index - 2);
923            end = end - 2;
924            uriBC.setEnd(end);
925        }
926 
927        index = 0;
928 
929        // Resolve occurrences of "/../" in the normalized path
930        while (true) {
931            index = uriBC.indexOf("/../", 0, 4, index);
932            if (index < 0)
933                break;
934            // Prevent from going outside our context
935            if (index == 0)
936                return false;
937            int index2 = -1;
938            for (pos = start + index - 1; (pos >= 0) && (index2 < 0); pos --) {
939                if (b[pos] == (byte) '/') {
940                    index2 = pos;
941                }
942            }
943            copyBytes(b, start + index2, start + index + 3,
944                      end - start - index - 3);
945            end = end + index2 - index - 3;
946            uriBC.setEnd(end);
947            index = index2;
948        }
949 
950        return true;
951 
952    }
