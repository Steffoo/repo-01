439     * Token can be read unambiguously with or without surrounding quotes so
440     * this parsing method for token permits optional surrounding double quotes.
441     * This is not defined in any RFC. It is a special case to handle data from
442     * buggy clients (known buggy clients for DIGEST auth include Microsoft IE 8
443     * &amp; 9, Apple Safari for OSX and iOS) that add quotes to values that
444     * should be tokens.
445     *
446     * @return the token if one was found, null if data other than a token or
447     *         quoted token was found or null if the end of data was reached
448     *         before a quoted token was terminated
449     */
450    private static String readQuotedToken(StringReader input)
451            throws IOException {
452 
453        StringBuilder result = new StringBuilder();
454        boolean quoted = false;
455 
456        int c = skipLws(input, false);
457 
458        if (c == '"') {
459            quoted = true;
460        } else if (c == -1 || !isToken(c)) {
461            return null;
462        } else {
463            result.append((char) c);
464        }
465        c = input.read();
466 
467        while (c != -1 && isToken(c)) {
468            result.append((char) c);
469            c = input.read();
470        }
471 
472        if (quoted) {
473            if (c != '"') {
474                return null;
475            }
476        } else {
477            // Skip back so non-token character is available for next read
478            input.skip(-1);
479        }
480 
481        if (c != -1 && result.length() == 0) {
482            return null;
483        } else {
484            return result.toString();
485        }
486    }

488    /**
489     * LHEX can be read unambiguously with or without surrounding quotes so this
490     * parsing method for LHEX permits optional surrounding double quotes. Some
491     * buggy clients (libwww-perl for DIGEST auth) are known to send quoted LHEX
492     * when the specification requires just LHEX.
493     *
494     * <p>
495     * LHEX are, literally, lower-case hexadecimal digits. This implementation
496     * allows for upper-case digits as well, converting the returned value to
497     * lower-case.
498     *
499     * @return  the sequence of LHEX (minus any surrounding quotes) if any was
500     *          found, or <code>null</code> if data other LHEX was found
501     */
502    private static String readLhex(StringReader input)
503            throws IOException {
504 
505        StringBuilder result = new StringBuilder();
506        boolean quoted = false;
507 
508        int c = skipLws(input, false);
509 
510        if (c == '"') {
511            quoted = true;
512        } else if (c == -1 || !isHex(c)) {
513            return null;
514        } else {
515            if ('A' <= c && c <= 'F') {
516                c -= ('A' - 'a');
517            }
518            result.append((char) c);
519        }
520        c = input.read();
521 
522        while (c != -1 && isHex(c)) {
523            if ('A' <= c && c <= 'F') {
524                c -= ('A' - 'a');
525            }
526            result.append((char) c);
527            c = input.read();
528        }
529 
530        if (quoted) {
531            if (c != '"') {
532                return null;
533            }
534        } else {
535            // Skip back so non-hex character is available for next read
536            input.skip(-1);
537        }
538 
539        if (c != -1 && result.length() == 0) {
540            return null;
541        } else {
542            return result.toString();
543        }
544    }
