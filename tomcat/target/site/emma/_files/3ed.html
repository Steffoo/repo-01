604    /**
605     * Map the specified URI.
606     */
607    private final void internalMap(CharChunk host, CharChunk uri,
608                                   MappingData mappingData)
609        throws Exception {
610 
611        if (mappingData.host != null) {
612            // The legacy code (dating down at least to Tomcat 4.1) just
613            // skipped all mapping work in this case. That behaviour has a risk
614            // of returning an inconsistent result.
615            // I do not see a valid use case for it.
616            throw new AssertionError();
617        }
618 
619        uri.setLimit(-1);
620 
621        Context[] contexts;
622        Context context = null;
623        int nesting = 0;
624 
625        // Virtual host mapping
626        Host[] hosts = this.hosts;
627        int pos = findIgnoreCase(hosts, host);
628        if ((pos != -1) && (host.equalsIgnoreCase(hosts[pos].name))) {
629            mappingData.host = hosts[pos].object;
630            contexts = hosts[pos].contextList.contexts;
631            nesting = hosts[pos].contextList.nesting;
632        } else {
633            if (defaultHostName == null) {
634                return;
635            }
636            pos = find(hosts, defaultHostName);
637            if ((pos != -1) && (defaultHostName.equals(hosts[pos].name))) {
638                mappingData.host = hosts[pos].object;
639                contexts = hosts[pos].contextList.contexts;
640                nesting = hosts[pos].contextList.nesting;
641            } else {
642                return;
643            }
644        }
645 
646        // Context mapping
647        pos = find(contexts, uri);
648        if (pos == -1) {
649            return;
650        }
651 
652        int lastSlash = -1;
653        int uriEnd = uri.getEnd();
654        int length = -1;
655        boolean found = false;
656        while (pos >= 0) {
657            if (uri.startsWith(contexts[pos].name)) {
658                length = contexts[pos].name.length();
659                if (uri.getLength() == length) {
660                    found = true;
661                    break;
662                } else if (uri.startsWithIgnoreCase("/", length)) {
663                    found = true;
664                    break;
665                }
666            }
667            if (lastSlash == -1) {
668                lastSlash = nthSlash(uri, nesting + 1);
669            } else {
670                lastSlash = lastSlash(uri);
671            }
672            uri.setEnd(lastSlash);
673            pos = find(contexts, uri);
674        }
675        uri.setEnd(uriEnd);
676 
677        if (!found) {
678            if (contexts[0].name.equals("")) {
679                context = contexts[0];
680            }
681        } else {
682            context = contexts[pos];
683        }
684        if (context != null) {
685            mappingData.context = context.object;
686            mappingData.contextPath.setString(context.name);
687        }
688 
689        // Wrapper mapping
690        if (context != null) {
691            internalMapWrapper(context, uri, mappingData);
692        }
693 
694    }

697    /**
698     * Wrapper mapping.
699     */
700    private final void internalMapWrapper(Context context, CharChunk path,
701                                          MappingData mappingData)
702        throws Exception {
703 
704        int pathOffset = path.getOffset();
705        int pathEnd = path.getEnd();
706        boolean noServletPath = false;
707 
708        int length = context.name.length();
709        if (length == (pathEnd - pathOffset)) {
710            noServletPath = true;
711        }
712        int servletPath = pathOffset + length;
713        path.setOffset(servletPath);
714 
715        // Rule 1 -- Exact Match
716        Wrapper[] exactWrappers = context.exactWrappers;
717        internalMapExactWrapper(exactWrappers, path, mappingData);
718 
719        // Rule 2 -- Prefix Match
720        boolean checkJspWelcomeFiles = false;
721        Wrapper[] wildcardWrappers = context.wildcardWrappers;
722        if (mappingData.wrapper == null) {
723            internalMapWildcardWrapper(wildcardWrappers, context.nesting, 
724                                       path, mappingData);
725            if (mappingData.wrapper != null && mappingData.jspWildCard) {
726                char[] buf = path.getBuffer();
727                if (buf[pathEnd - 1] == '/') {
728                    /*
729                     * Path ending in '/' was mapped to JSP servlet based on
730                     * wildcard match (e.g., as specified in url-pattern of a
731                     * jsp-property-group.
732                     * Force the context's welcome files, which are interpreted
733                     * as JSP files (since they match the url-pattern), to be
734                     * considered. See Bugzilla 27664.
735                     */ 
736                    mappingData.wrapper = null;
737                    checkJspWelcomeFiles = true;
738                } else {
739                    // See Bugzilla 27704
740                    mappingData.wrapperPath.setChars(buf, path.getStart(),
741                                                     path.getLength());
742                    mappingData.pathInfo.recycle();
743                }
744            }
745        }
746 
747        if(mappingData.wrapper == null && noServletPath &&
748                context.mapperContextRootRedirectEnabled) {
749            // The path is empty, redirect to "/"
750            path.append('/');
751            pathEnd = path.getEnd();
752            mappingData.redirectPath.setChars
753                (path.getBuffer(), pathOffset, pathEnd - pathOffset);
754            path.setEnd(pathEnd - 1);
755            return;
756        }
757 
758        // Rule 3 -- Extension Match
759        Wrapper[] extensionWrappers = context.extensionWrappers;
760        if (mappingData.wrapper == null && !checkJspWelcomeFiles) {
761            internalMapExtensionWrapper(extensionWrappers, path, mappingData);
762        }
763 
764        // Rule 4 -- Welcome resources processing for servlets
765        if (mappingData.wrapper == null) {
766            boolean checkWelcomeFiles = checkJspWelcomeFiles;
767            if (!checkWelcomeFiles) {
768                char[] buf = path.getBuffer();
769                checkWelcomeFiles = (buf[pathEnd - 1] == '/');
770            }
771            if (checkWelcomeFiles) {
772                for (int i = 0; (i < context.welcomeResources.length)
773                         && (mappingData.wrapper == null); i++) {
774                    path.setOffset(pathOffset);
775                    path.setEnd(pathEnd);
776                    path.append(context.welcomeResources[i], 0,
777                                context.welcomeResources[i].length());
778                    path.setOffset(servletPath);
779 
780                    // Rule 4a -- Welcome resources processing for exact macth
781                    internalMapExactWrapper(exactWrappers, path, mappingData);
782 
783                    // Rule 4b -- Welcome resources processing for prefix match
784                    if (mappingData.wrapper == null) {
785                        internalMapWildcardWrapper
786                            (wildcardWrappers, context.nesting, 
787                             path, mappingData);
788                    }
789 
790                    // Rule 4c -- Welcome resources processing
791                    //            for physical folder
792                    if (mappingData.wrapper == null
793                        && context.resources != null) {
794                        Object file = null;
795                        String pathStr = path.toString();
796                        try {
797                            file = context.resources.lookup(pathStr);
798                        } catch(NamingException nex) {
799                            // Swallow not found, since this is normal
800                        }
801                        if (file != null && !(file instanceof DirContext) ) {
802                            internalMapExtensionWrapper(extensionWrappers,
803                                                        path, mappingData);
804                            if (mappingData.wrapper == null
805                                && context.defaultWrapper != null) {
806                                mappingData.wrapper =
807                                    context.defaultWrapper.object;
808                                mappingData.requestPath.setChars
809                                    (path.getBuffer(), path.getStart(), 
810                                     path.getLength());
811                                mappingData.wrapperPath.setChars
812                                    (path.getBuffer(), path.getStart(), 
813                                     path.getLength());
814                                mappingData.requestPath.setString(pathStr);
815                                mappingData.wrapperPath.setString(pathStr);
816                            }
817                        }
818                    }
819                }
820 
821                path.setOffset(servletPath);
822                path.setEnd(pathEnd);
823            }
824                                        
825        }
826 
827 
828        // Rule 7 -- Default servlet
829        if (mappingData.wrapper == null && !checkJspWelcomeFiles) {
830            if (context.defaultWrapper != null) {
831                mappingData.wrapper = context.defaultWrapper.object;
832                mappingData.requestPath.setChars
833                    (path.getBuffer(), path.getStart(), path.getLength());
834                mappingData.wrapperPath.setChars
835                    (path.getBuffer(), path.getStart(), path.getLength());
836            }
837            // Redirection to a folder
838            char[] buf = path.getBuffer();
839            if (context.resources != null && buf[pathEnd -1 ] != '/') {
840                Object file = null;
841                String pathStr = path.toString();
842                try {
843                    if (pathStr.length() == 0) {
844                        file = context.resources.lookup("/");
845                    } else {
846                        file = context.resources.lookup(pathStr);
847                    }
848                } catch(NamingException nex) {
849                    // Swallow, since someone else handles the 404
850                }
851                if (file != null && file instanceof DirContext &&
852                        context.mapperDirectoryRedirectEnabled) {
853                    // Note: this mutates the path: do not do any processing 
854                    // after this (since we set the redirectPath, there 
855                    // shouldn't be any)
856                    path.setOffset(pathOffset);
857                    path.append('/');
858                    mappingData.redirectPath.setChars
859                        (path.getBuffer(), path.getStart(), path.getLength());
860                } else {
861                    mappingData.requestPath.setString(pathStr);
862                    mappingData.wrapperPath.setString(pathStr);
863                }
864            }
865        }
866 
867        path.setOffset(pathOffset);
868        path.setEnd(pathEnd);
869    }
