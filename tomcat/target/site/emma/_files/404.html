506     * Similar to a String.IndexOf, but uses pure bytes
507     * @param src - the source bytes to be searched
508     * @param srcOff - offset on the source buffer
509     * @param find - the string to be found within src
510     * @return - the index of the first matching byte. -1 if the find array is not found
511     */
512    public static int firstIndexOf(byte[] src, int srcOff, byte[] find){
513        int result = -1;
514        if (find.length > src.length) return result;
515        if (find.length == 0 || src.length == 0) return result;
516        if (srcOff >= src.length ) throw new java.lang.ArrayIndexOutOfBoundsException();
517        boolean found = false;
518        int srclen = src.length;
519        int findlen = find.length;
520        byte first = find[0];
521        int pos = srcOff;
522        while (!found) {
523            //find the first byte
524            while (pos < srclen){
525                if (first == src[pos])
526                    break;
527                pos++;
528            }
529            if (pos >= srclen)
530                return -1;
531 
532            //we found the first character
533            //match the rest of the bytes - they have to match
534            if ( (srclen - pos) < findlen)
535                return -1;
536            //assume it does exist
537            found = true;
538            for (int i = 1; ( (i < findlen) && found); i++)
539                found = found && (find[i] == src[pos + i]);
540            if (found)
541                result = pos;
542            else if ( (srclen - pos) < findlen)
543                return -1; //no more matches possible
544            else
545                pos++;
546        }
547        return result;
548    }
