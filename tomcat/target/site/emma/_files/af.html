529        public boolean parse(Request request, String authorization) {
530            // Validate the authorization credentials format
531            if (authorization == null) {
532                return false;
533            }
534            if (!authorization.startsWith("Digest ")) {
535                return false;
536            }
537            authorization = authorization.substring(7).trim();
538 
539            // Bugzilla 37132: http://bz.apache.org/bugzilla/show_bug.cgi?id=37132
540            String[] tokens = authorization.split(",(?=(?:[^\"]*\"[^\"]*\")+$)");
541 
542            method = request.getMethod();
543 
544            for (int i = 0; i < tokens.length; i++) {
545                String currentToken = tokens[i];
546                if (currentToken.length() == 0)
547                    continue;
548 
549                int equalSign = currentToken.indexOf('=');
550                if (equalSign < 0) {
551                    return false;
552                }
553                String currentTokenName =
554                    currentToken.substring(0, equalSign).trim();
555                String currentTokenValue =
556                    currentToken.substring(equalSign + 1).trim();
557                if ("username".equals(currentTokenName))
558                    userName = removeQuotes(currentTokenValue);
559                if ("realm".equals(currentTokenName))
560                    realmName = removeQuotes(currentTokenValue, true);
561                if ("nonce".equals(currentTokenName))
562                    nonce = removeQuotes(currentTokenValue);
563                if ("nc".equals(currentTokenName))
564                    nc = removeQuotes(currentTokenValue);
565                if ("cnonce".equals(currentTokenName))
566                    cnonce = removeQuotes(currentTokenValue);
567                if ("qop".equals(currentTokenName))
568                    qop = removeQuotes(currentTokenValue);
569                if ("uri".equals(currentTokenName))
570                    uri = removeQuotes(currentTokenValue);
571                if ("response".equals(currentTokenName))
572                    response = removeQuotes(currentTokenValue);
573                if ("opaque".equals(currentTokenName))
574                    opaqueReceived = removeQuotes(currentTokenValue);
575            }
576 
577            return true;
578        }

580        public boolean validate(Request request, LoginConfig config) {
581            if ( (userName == null) || (realmName == null) || (nonce == null)
582                 || (uri == null) || (response == null) ) {
583                return false;
584            }
585 
586            // Validate the URI - should match the request line sent by client
587            if (validateUri) {
588                String uriQuery;
589                String query = request.getQueryString();
590                if (query == null) {
591                    uriQuery = request.getRequestURI();
592                } else {
593                    uriQuery = request.getRequestURI() + "?" + query;
594                }
595                if (!uri.equals(uriQuery)) {
596                    // Some clients (older Android) use an absolute URI for
597                    // DIGEST but a relative URI in the request line.
598                    // request. 2.3.5 < fixed Android version <= 4.0.3
599                    String host = request.getHeader("host");
600                    String scheme = request.getScheme();
601                    if (host != null && !uriQuery.startsWith(scheme)) {
602                        StringBuilder absolute = new StringBuilder();
603                        absolute.append(scheme);
604                        absolute.append("://");
605                        absolute.append(host);
606                        absolute.append(uriQuery);
607                        if (!uri.equals(absolute.toString())) {
608                            return false;
609                        }
610                    } else {
611                        return false;
612                    }
613                }
614            }
615 
616            // Validate the Realm name
617            String lcRealm = config.getRealmName();
618            if (lcRealm == null) {
619                lcRealm = REALM_NAME;
620            }
621            if (!lcRealm.equals(realmName)) {
622                return false;
623            }
624            
625            // Validate the opaque string
626            if (!opaque.equals(opaqueReceived)) {
627                return false;
628            }
629 
630            // Validate nonce
631            int i = nonce.indexOf(":");
632            if (i < 0 || (i + 1) == nonce.length()) {
633                return false;
634            }
635            long nonceTime;
636            try {
637                nonceTime = Long.parseLong(nonce.substring(0, i));
638            } catch (NumberFormatException nfe) {
639                return false;
640            }
641            String md5clientIpTimeKey = nonce.substring(i + 1);
642            long currentTime = System.currentTimeMillis();
643            if ((currentTime - nonceTime) > nonceValidity) {
644                nonceStale = true;
645                synchronized (nonces) {
646                    nonces.remove(nonce);
647                }
648            }
649            String serverIpTimeKey =
650                request.getRemoteAddr() + ":" + nonceTime + ":" + key;
651            byte[] buffer = null;
652            synchronized (md5Helper) {
653                buffer = md5Helper.digest(serverIpTimeKey.getBytes());
654            }
655            String md5ServerIpTimeKey = md5Encoder.encode(buffer);
656            if (!md5ServerIpTimeKey.equals(md5clientIpTimeKey)) {
657                return false;
658            }
659 
660            // Validate qop
661            if (qop != null && !QOP.equals(qop)) {
662                return false;
663            }
664 
665            // Validate cnonce and nc
666            // Check if presence of nc and Cnonce is consistent with presence of qop
667            if (qop == null) {
668                if (cnonce != null || nc != null) {
669                    return false;
670                }
671            } else {
672                if (cnonce == null || nc == null) {
673                    return false;
674                }
675                // RFC 2617 says nc must be 8 digits long. Older Android clients
676                // use 6. 2.3.5 < fixed Android version <= 4.0.3
677                if (nc.length() < 6 || nc.length() > 8) {
678                    return false;
679                }
680                long count;
681                try {
682                    count = Long.parseLong(nc, 16);
683                } catch (NumberFormatException nfe) {
684                    return false;
685                }
686                NonceInfo info;
687                synchronized (nonces) {
688                    info = nonces.get(nonce);
689                }
690                if (info == null) {
691                    // Nonce is valid but not in cache. It must have dropped out
692                    // of the cache - force a re-authentication
693                    nonceStale = true;
694                } else {
695                    if (!info.nonceCountValid(count)) {
696                        return false;
697                    }
698                }
699            }
700            return true;
701        }
