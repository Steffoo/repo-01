845     * Invalidate HttpSessions
846     * @param sessionIds
847     * @return number of invalidated sessions
848     * @throws IOException 
849     */
850    public int invalidateSessions(String path, String[] sessionIds) throws IOException {
851        if (null == sessionIds) {
852            return 0;
853        }
854        int nbAffectedSessions = 0;
855        for (int i = 0; i < sessionIds.length; ++i) {
856            String sessionId = sessionIds[i];
857            HttpSession session = getSessionForPathAndId(path, sessionId).getSession();
858            if (null == session) {
859                // Shouldn't happen, but let's play nice...
860                    if (debug >= 1) {
861                            log("WARNING: can't invalidate null session " + sessionId);
862                    }
863                continue;
864            }
865            try {
