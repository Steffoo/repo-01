122        pushOpp(null);

124        while (et.hasMoreTokens()) {
125            int token = et.nextToken();
126            if (token != ExpressionTokenizer.TOKEN_STRING)
127                currStringNode = null;
128            switch (token) {

132                        nodeStack.add(0, currStringNode);

135                        currStringNode.value.append(" ");
136                        currStringNode.value.append(et.getTokenValue());
137                    }
138                    break;
139                case ExpressionTokenizer.TOKEN_AND :
140                    pushOpp(new AndNode());
141                    break;
142                case ExpressionTokenizer.TOKEN_OR :
143                    pushOpp(new OrNode());
144                    break;
145                case ExpressionTokenizer.TOKEN_NOT :
146                    pushOpp(new NotNode());

150                    break;

154                    // be resolved when the next opp comes along.
155                    oppStack.add(0, new EqualNode());
156                    break;
157                case ExpressionTokenizer.TOKEN_RBRACE :

159                    resolveGroup();

161                case ExpressionTokenizer.TOKEN_LBRACE :

163                    pushOpp(null);
164                    break;
165                case ExpressionTokenizer.TOKEN_GE :
166                    pushOpp(new NotNode());

168                    // is NOT less than

170                    break;
171                case ExpressionTokenizer.TOKEN_LE :
172                    pushOpp(new NotNode());
173                    // Similar stategy to NOT_EQ above, except this
174                    // is NOT greater than

176                    break;
177                case ExpressionTokenizer.TOKEN_GT :
178                    pushOpp(new GreaterThanNode());
179                    break;
180                case ExpressionTokenizer.TOKEN_LT :
181                    pushOpp(new LessThanNode());
182                    break;
183                case ExpressionTokenizer.TOKEN_END :
184                    break;
185            }
186        }
187        // Finish off the rest of the opps
188        resolveGroup();

191        }

199    }
